20   package net.tharow.tantalum.platform.io;
21   
22   import net.tharow.tantalum.launchercore.TantalumConstants;
23   
24   import java.util.Date;
25   import java.util.Locale;
26   
27   import static net.tharow.tantalum.utilslib.MD5Utils.getMD5;
28   
29   @SuppressWarnings({"unused"})
30   public class FeedItem {
31       private String user;
32       private long date;
33       private String content;
34       private String avatar;
35       private String url;
36   
37       public FeedItem() {
38       }
39   
40       public String getUser() {
41           return user!=null?user:TantalumConstants.SYSTEM_USERNAME;
42       }
43   
44       public Date getDate() {
45           return new Date(date * 1000);
46       }
47   
48       public String getContent() {
49           return content;
50       }
51   
52       public String getAvatar() {
53           if (this.avatar==null || this.avatar.isEmpty()){
54               return "https://www.gravatar.com/avatar/" + getMD5(this.user.toLowerCase(Locale.ROOT)) + "?d=identicon&s=2048";
55           }
56           return this.avatar;
57       }
58   
59       public String getUrl() {
60           return url;
61       }
62   
63       public AuthorshipInfo getAuthorship() {
64           return new AuthorshipInfo(getUser(), getAvatar(), getDate());
65       }
66   }
