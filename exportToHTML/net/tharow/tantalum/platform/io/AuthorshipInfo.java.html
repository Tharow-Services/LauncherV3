20   package net.tharow.tantalum.platform.io;
21   
22   import net.tharow.tantalum.launchercore.TantalumConstants;
23   
24   import java.util.Date;
25   
26   @SuppressWarnings({"unused"})
27   public class AuthorshipInfo {
28       private String user;
29       private String avatar;
30       private Date date;
31   
32       public AuthorshipInfo() {
33       }
34   
35       public AuthorshipInfo(String user, String avatar, Date date) {
36           this.user = user;
37           this.avatar = avatar;
38           this.date = date;
39       }
40   
41       public String getUser() {
42           return user!=null?user:TantalumConstants.SYSTEM_USERNAME;
43       }
44   
45       public String getAvatar() {
46           return this.avatar!=null?this.avatar:TantalumConstants.AVATAR_URL+this.getUser()+".png";
47       }
48   
49       public Date getDate() {
50           return date;
51       }
52   }
