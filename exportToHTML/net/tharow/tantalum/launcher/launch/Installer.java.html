92       public void cancel() {
93           Utils.getLogger().info("User pressed cancel button.");
94           synchronized (cancelLock) {
95               isCancelledByUser = true;
96           }
97           runningThread.interrupt();
98       }

108      protected void internalInstallAndRun(final ResourceLoader resources, final ModpackModel pack, final String build, final boolean doFullInstall, final LauncherFrame frame, final DownloadListener listener, final boolean doLaunch) {
109          runningThread = new Thread(() -> {
110              boolean everythingWorked = false;
111  
112              try {
113                  MojangVersion version;
114  
115                  InstallTasksQueue<MojangVersion> tasksQueue = new InstallTasksQueue<>(listener);
116                  MojangVersionBuilder versionBuilder = createVersionBuilder(pack, tasksQueue);
117  
118                  if (!pack.isLocalOnly() && build != null && !build.isEmpty()) {
119                      buildTasksQueue(tasksQueue, resources, pack, build, doFullInstall, versionBuilder);
120  
121                      version = installer.installPack(tasksQueue, pack, build);
122                  } else {
123                      version = versionBuilder.buildVersionFromKey(null);
124  
125                      if (version != null)
126                          pack.initDirectories();
127                  }
128  
129                  if (doLaunch) {
130                      if (version == null) {
131                          throw new PackNotAvailableOfflineException(pack.getDisplayName());
132                      }
133  
134                      boolean usingMojangJava = version.getJavaVersion() != null && settings.shouldUseMojangJava();
135  
136                      JavaVersionRepository javaVersions = launcher.getJavaVersions();
137                      Memory memoryObj = Memory.getClosestAvailableMemory(Memory.getMemoryFromId(settings.getMemory()), javaVersions.getSelectedVersion().is64Bit());
138                      long memory = memoryObj.getMemoryMB();
139                      String versionNumber = javaVersions.getSelectedVersion().getVersionNumber();
140                      RunData data = pack.getRunData();
141  
142                      if (data != null && !data.isRunDataValid(memory, versionNumber, usingMojangJava)) {
143                          FixRunDataDialog dialog = new FixRunDataDialog(frame, resources, data, javaVersions, memoryObj, settings.shouldAutoAcceptModpackRequirements(), usingMojangJava);
144                          dialog.setVisible(true);
145                          if (dialog.getResult() == FixRunDataDialog.Result.ACCEPT) {
146                              memoryObj = dialog.getRecommendedMemory();
147                              memory = memoryObj.getMemoryMB();
148                              IJavaVersion recommendedJavaVersion = dialog.getRecommendedJavaVersion();
149                              javaVersions.selectVersion(recommendedJavaVersion.getVersionNumber(), recommendedJavaVersion.is64Bit());
150  
151                              if (dialog.shouldRemember()) {
152                                  settings.setAutoAcceptModpackRequirements(true);
153                              }
154                          } else
155                              return;
156                      }
157  
158                      if (!usingMojangJava && RunData.isJavaVersionAtLeast(versionNumber, "1.9")) {
159                          int result = JOptionPane.showConfirmDialog(frame, resources.getString("launcher.jverwarning", versionNumber), resources.getString("launcher.jverwarning.title"), JOptionPane.YES_NO_OPTION, JOptionPane.WARNING_MESSAGE);
160                          if (result != JOptionPane.YES_OPTION)
161                              return;
162                      }
163  
164                      LaunchAction launchAction = settings.getLaunchAction();
165  
166                      if (launchAction == null || launchAction == LaunchAction.HIDE) {
167                          launcherUnhider = new LauncherUnhider(settings, frame);
168                      } else
169                          launcherUnhider = null;
170  
171                      LaunchOptions options = new LaunchOptions(pack.getDisplayName(), packIconMapper.getImageLocation(pack).getAbsolutePath(), settings);
172                      launcher.launch(pack, memory, options, launcherUnhider, version);
173  
174                      if (launchAction == null || launchAction == LaunchAction.HIDE) {
175                          frame.setVisible(false);
176                      } else if (launchAction == LaunchAction.NOTHING) {
177                          EventQueue.invokeLater(frame::launchCompleted);
178                      } else if (launchAction == LaunchAction.CLOSE) {
179                          System.exit(0);
180                      }
181                  }
182  
183                  everythingWorked = true;
184              } catch (InterruptedException e) {
185                  boolean cancelledByUser = false;
186                  synchronized (cancelLock) {
187                      if (isCancelledByUser) {
188                          cancelledByUser = true;
189                          isCancelledByUser = false;
190                      }
191                  }
192  
193                  //Canceled by user
194                  if (!cancelledByUser) {
195                      if (e.getCause() != null)
196                          Utils.getLogger().info("Cancelled by exception.");
197                      else
198                          Utils.getLogger().info("Cancelled by code.");
199                      e.printStackTrace();
200                  } else
201                      Utils.getLogger().info("Cancelled by user.");
202              } catch (PackNotAvailableOfflineException e) {
203                  JOptionPane.showMessageDialog(frame, e.getMessage(), resources.getString("launcher.installerror.unavailable"), JOptionPane.WARNING_MESSAGE);
204              } catch (DownloadException e) {
205                  JOptionPane.showMessageDialog(frame, resources.getString("launcher.installerror.download", pack.getDisplayName(), e.getMessage()), resources.getString("launcher.installerror.title"), JOptionPane.WARNING_MESSAGE);
206              } catch (ZipException e) {
207                  JOptionPane.showMessageDialog(frame, resources.getString("launcher.installerror.unzip", pack.getDisplayName(), e.getMessage()), resources.getString("launcher.installerror.title"), JOptionPane.WARNING_MESSAGE);
208              } catch (CacheDeleteException e) {
209                  JOptionPane.showMessageDialog(frame, resources.getString("launcher.installerror.cache", pack.getDisplayName(), e.getMessage()), resources.getString("launcher.installerror.title"), JOptionPane.WARNING_MESSAGE);
210              } catch (BuildInaccessibleException e) {
211                  e.printStackTrace();
212                  JOptionPane.showMessageDialog(frame, e.getMessage(), resources.getString("launcher.installerror.title"), JOptionPane.WARNING_MESSAGE);
213              } catch (Exception e) {
214                  e.printStackTrace();
215              } finally {
216                  if (!everythingWorked || !doLaunch) {
217                      EventQueue.invokeLater(frame::launchCompleted);
218                  }
219              }
220          }) {
221              ///Interrupt is being called from a mysterious source, so unless this is a user-initiated cancel
222              ///Let's print the stack trace of the interrupter.
223              @Override
224              public void interrupt() {
225                  boolean userCancelled = false;
226                  synchronized (cancelLock) {
227                      if (isCancelledByUser)
228                          userCancelled = true;
229                  }
230  
231                  if (!userCancelled) {
232                      Utils.getLogger().info("Mysterious interruption source.");
233                      try {
234                          //I am a charlatan and a hack.
235                          throw new Exception("Grabbing stack trace- this isn't necessarily an error.");
236                      } catch (Exception ex) {
237                          ex.printStackTrace();
238                      }
239                  }
240                  super.interrupt();
241              }
242          };
243          runningThread.start();
244      }

246      public boolean isCurrentlyRunning() {
247          if (runningThread != null && runningThread.isAlive())
248              return true;
249          return launcherUnhider != null && !launcherUnhider.hasExited();
250      }

252      public void buildTasksQueue(InstallTasksQueue<? extends MojangVersion> queue, ResourceLoader resources, ModpackModel modpack, String build, boolean doFullInstall, MojangVersionBuilder versionBuilder) throws CacheDeleteException, BuildInaccessibleException {
253          PackInfo packInfo = modpack.getPackInfo();
254          Modpack modpackData = packInfo.getModpack(build);
255  
256          if (modpackData.getGameVersion() == null)
257              return;
258  
259          String minecraft = modpackData.getGameVersion();
260          Version installedVersion = modpack.getInstalledVersion();
261  
262          TaskGroup examineModpackData = new TaskGroup(resources.getString("install.message.examiningmodpack"));
263          TaskGroup verifyingFiles = new TaskGroup(resources.getString("install.message.verifyingfiles"));
264          TaskGroup downloadingMods = new TaskGroup(resources.getString("install.message.downloadmods"));
265          TaskGroup installingMods = new TaskGroup(resources.getString("install.message.installmods"));
266          TaskGroup checkVersionFile = new TaskGroup(resources.getString("install.message.checkversionfile"));
267          TaskGroup installVersionFile = new TaskGroup(resources.getString("install.message.installversionfile"));
268          TaskGroup rundataTaskGroup = new TaskGroup(resources.getString("install.message.runData"));
269          TaskGroup examineVersionFile = new TaskGroup(resources.getString("install.message.examiningversionfile"));
270          TaskGroup grabLibs = new TaskGroup(resources.getString("install.message.grablibraries"));
271          TaskGroup checkNonMavenLibs = new TaskGroup(resources.getString("install.message.nonmavenlibs"));
272          TaskGroup installingLibs = new TaskGroup(resources.getString("install.message.installlibs"));
273          TaskGroup installingMinecraft = new TaskGroup(resources.getString("install.message.installminecraft"));
274          TaskGroup examineIndex = new TaskGroup(resources.getString("install.message.examiningindex"));
275          TaskGroup verifyingAssets = new TaskGroup(resources.getString("install.message.verifyassets"));
276          TaskGroup installingAssets = new TaskGroup(resources.getString("install.message.installassets"));
277          TaskGroup examineJava = new TaskGroup("Examining Java runtime...");
278          TaskGroup downloadJava = new TaskGroup("Downloading Java runtime...");
279  
280          queue.addTask(examineModpackData);
281          queue.addTask(verifyingFiles);
282          queue.addTask(downloadingMods);
283          queue.addTask(installingMods);
284          queue.addTask(checkVersionFile);
285          queue.addTask(installVersionFile);
286          queue.addTask(rundataTaskGroup);
287          queue.addTask(examineVersionFile);
288          queue.addTask(grabLibs);
289          queue.addTask(checkNonMavenLibs);
290          queue.addTask(installingLibs);
291          queue.addTask(installingMinecraft);
292          queue.addTask(examineIndex);
293          queue.addTask(verifyingAssets);
294          queue.addTask(installingAssets);
295          queue.addTask(examineJava);
296          queue.addTask(downloadJava);
297          if (OperatingSystem.getOperatingSystem() == OperatingSystem.OSX)
298              queue.addTask(new CopyDylibJnilibTask(modpack));
299  
300          // Add FML libs
301          String fmlLibsZip;
302          File modpackFmlLibDir = new File(modpack.getInstalledDirectory(), "lib");
303          HashMap<String, String> fmlLibs = new HashMap<>();
304  
305          switch (minecraft) {
306              /*case "1.4", "1.4.1", "1.4.2", "1.4.3", "1.4.4", "1.4.5", "1.4.6", "1.4.7" -> fmlLibsZip = "fml_libs.zip"; 
307              case "1.5" -> { 
308                  fmlLibsZip = "fml_libs15.zip"; 
309                  fmlLibs.put("deobfuscation_data_1.5.zip", "dba6d410a91a855f3b84457c86a8132a"); 
310              } 
311              case "1.5.1" -> { 
312                  fmlLibsZip = "fml_libs15.zip"; 
313                  fmlLibs.put("deobfuscation_data_1.5.1.zip", "c4fc2fedba60d920e4c7f9a095b2b883"); 
314              }*/
315              case "1.5.2" : {
316                  fmlLibsZip = "fml_libs15.zip";
317                  fmlLibs.put("deobfuscation_data_1.5.2.zip", "270d9775872cc9fa773389812cab91fe");
318              }
319              default : fmlLibsZip = "";
320          }
321  
322          if (!fmlLibsZip.isEmpty()) {
323              verifyingFiles.addTask(new EnsureFileTask<>(new File(directories.getCacheDirectory(), fmlLibsZip), new ValidZipFileVerifier(), modpackFmlLibDir, TantalumConstants.FML_REPO_URL + fmlLibsZip, installingLibs, installingLibs));
324          }
325  
326          if (!fmlLibs.isEmpty()) {
327              fmlLibs.forEach((name, md5) -> {
328                  MD5FileVerifier verifier = null;
329  
330                  if (!md5.isEmpty())
331                      verifier = new MD5FileVerifier(md5);
332  
333                  File cached = new File(directories.getCacheDirectory(), name);
334                  File target = new File(modpackFmlLibDir, name);
335  
336                  if (!target.exists() || (verifier != null && !verifier.isFileValid(target)) ) {
337                      verifyingFiles.addTask(new EnsureFileTask<>(cached, verifier, null, TantalumConstants.FML_REPO_URL + name, installingLibs, installingLibs));
338                      installingLibs.addTask(new CopyFileTask(cached, target));
339                  }
340              });
341          }
342  
343          if (doFullInstall) {
344              //If we're installing a new version of modpack, then we need to get rid of the existing version.json
345              File versionFile = new File(modpack.getBinDir(), "version.json");
346              if (versionFile.exists()) {
347                  if (!versionFile.delete()) {
348                      throw new CacheDeleteException(versionFile.getAbsolutePath());
349                  }
350              }
351  
352              // Remove bin/install_profile.json, which is used by ForgeWrapper to install Forge in Minecraft 1.13+
353              // (and the latest few Forge builds in 1.12.2)
354              File installProfileFile = new File(modpack.getBinDir(), "install_profile.json");
355              if (installProfileFile.exists()) {
356                  if (!installProfileFile.delete()) {
357                      throw new CacheDeleteException(installProfileFile.getAbsolutePath());
358                  }
359              }
360  
361              // Delete all other version JSON files in the bin dir
362              File[] binFiles = modpack.getBinDir().listFiles();
363              if (binFiles != null) {
364                  final Pattern minecraftVersionPattern = Pattern.compile("^[0-9]+(\\.[0-9]+)+\\.json$");
365                  for (File binFile : binFiles) {
366                      if (minecraftVersionPattern.matcher(binFile.getName()).matches()) {
367                          if (!binFile.delete()) {
368                              throw new CacheDeleteException(binFile.getAbsolutePath());
369                          }
370                      }
371                  }
372              }
373  
374              // Remove the runData file between updates/reinstall as well
375              File runData = new File(modpack.getBinDir(), "runData");
376              if (runData.exists()) {
377                  if (!runData.delete()) {
378                      throw new CacheDeleteException(runData.getAbsolutePath());
379                  }
380              }
381  
382              // Remove the bin/modpack.jar file
383              // This prevents issues when upgrading a modpack between a version that has a modpack.jar, and
384              // one that doesn't. One example of this is updating BareBonesPack from a Forge to a Fabric build.
385              File modpackJar = new File(modpack.getBinDir(), "modpack.jar");
386              if (modpackJar.exists()) {
387                  if (!modpackJar.delete()) {
388                      throw new CacheDeleteException(modpackJar.getAbsolutePath());
389                  }
390              }
391  
392              examineModpackData.addTask(new CleanupAndExtractModpackTask(modpack, modpackData, verifyingFiles, downloadingMods, installingMods));
393          }
394  
395          if (doFullInstall)
396              rundataTaskGroup.addTask(new WriteRundataFile(modpack, modpackData));
397          else
398              rundataTaskGroup.addTask(new CheckRundataFile(modpack, modpackData, rundataTaskGroup));
399  
400          checkVersionFile.addTask(new VerifyVersionFilePresentTask(modpack, minecraft, versionBuilder));
401          examineVersionFile.addTask(new HandleVersionFileTask(modpack, directories, checkNonMavenLibs, grabLibs, installingLibs, installingLibs, versionBuilder));
402          examineVersionFile.addTask(new EnsureAssetsIndexTask(directories.getAssetsDirectory(), modpack, installingMinecraft, examineIndex, verifyingAssets, installingAssets, installingAssets));
403          if(settings.shouldUseMojangJava()){
404              examineJava.addTask(new EnsureJavaRuntimeManifestTask(directories.getRuntimesDirectory(), modpack, examineJava, downloadJava));
405          }
406  
407          if (doFullInstall || (installedVersion != null && installedVersion.isLegacy()))
408              installingMinecraft.addTask(new InstallMinecraftIfNecessaryTask(modpack, minecraft, directories.getCacheDirectory()));
409      }

411      private MojangVersionBuilder createVersionBuilder(ModpackModel modpack, InstallTasksQueue<? extends MojangVersion> tasksQueue) {
412  
413          ZipFileRetriever zipVersionRetriever = new ZipFileRetriever(new File(modpack.getBinDir(), "modpack.jar"));
414          HttpFileRetriever fallbackVersionRetriever = new HttpFileRetriever(TantalumConstants.VERSIONS_URL, tasksQueue.getDownloadListener());
415  
416          ArrayList<MojangVersionRetriever> fallbackRetrievers = new ArrayList<>(1);
417          fallbackRetrievers.add(fallbackVersionRetriever);
418  
419          File versionJson = new File(modpack.getBinDir(), "version.json");
420  
421          // This always gets the version.json from the modpack.jar (it ignores "key"), cached as bin/version.json
422          FileVersionBuilder zipVersionBuilder = new FileVersionBuilder(versionJson, zipVersionRetriever, fallbackRetrievers);
423          // This gets the "key" from bin/$key.json if it exists, otherwise it downloads it from our repo into that location
424          FileVersionBuilder webVersionBuilder = new FileVersionBuilder(modpack.getBinDir(), null, fallbackRetrievers);
425  
426          return new ChainVersionBuilder(zipVersionBuilder, webVersionBuilder);
427      }
