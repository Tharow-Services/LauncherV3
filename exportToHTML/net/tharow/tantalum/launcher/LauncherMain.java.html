152      public static void main(String[] argv) {
153          try {
154              UIManager.setLookAndFeel(UIManager.getSystemLookAndFeelClassName());
155          } catch (Exception ex) {
156              Utils.getLogger().log(Level.SEVERE, ex.getMessage(), ex);
157          }
158  
159          ToolTipManager.sharedInstance().setDismissDelay(Integer.MAX_VALUE);
160  
161          StartupParameters params = new StartupParameters(argv);
162          try {
163              JCommander.newBuilder()
164                      .addObject(params)
165                      .build()
166                      .parse(argv);
167          } catch (Exception ex) {
168              ex.printStackTrace();
169          }
170  
171          TantalumSettings settings = null;
172  
173          try {
174              settings = SettingsFactory.buildSettingsObject(Relauncher.getRunningPath(LauncherMain.class), params.isMover());
175          } catch (UnsupportedEncodingException ex) {
176              ex.printStackTrace();
177          }
178  
179          if (settings == null) {
180              ResourceLoader installerResources = new ResourceLoader(null, "net","tharow","tantalum","launcher","resources");
181              installerResources.setSupportedLanguages(supportedLanguages);
182              installerResources.setLocale(ResourceLoader.DEFAULT_LOCALE);
183              InstallerFrame dialog = new InstallerFrame(installerResources, params);
184              dialog.setVisible(true);
185              return;
186          }
187  
188          LauncherDirectories directories = new TantalumLauncherDirectories(settings.getTechnicRoot());
189          ResourceLoader resources = new ResourceLoader(directories, "net","tharow","tantalum","launcher","resources");
190          resources.setSupportedLanguages(supportedLanguages);
191          resources.setLocale(settings.getLanguageCode());
192  
193          // Sanity check
194          checkIfRunningInsideOneDrive(directories.getLauncherDirectory());
195  
196          if (params.getBuildNumber() != null && !params.getBuildNumber().isEmpty()) {
197              buildNumber = new CommandLineBuildNumber(params);
198          } else {
199              buildNumber = new VersionFileBuildNumber(resources);
200          }
201          File unlockedFile = new File(directories.getLauncherDirectory(), "Unlocked-Mode.lock");
202          if (params.isUnlocked() && unlockedFile.exists()) {
203              //noinspection ResultOfMethodCallIgnored
204              unlockedFile.delete();
205          }
206          if (params.isUnlocked() && !unlockedFile.exists()) {
207              try {
208                  checkUnlockCode();
209                  //noinspection ResultOfMethodCallIgnored
210                  unlockedFile.createNewFile();
211              } catch (IOException | AuthenticationException ignored) {}
212          }
213          TantalumConstants.setUnlocked(unlockedFile.exists());
214          TantalumConstants.setBuildNumber(buildNumber);
215  
216          setupLogging(directories, resources);
217  
218  
219          //Currently Unused
220          //String launcherBuild = buildNumber.getBuildNumber();
221          int build = -1;
222  
223          try {
224              build = Integer.parseInt((new VersionFileBuildNumber(resources)).getBuildNumber());
225          } catch (NumberFormatException | StringIndexOutOfBoundsException ex) {
226              //This is probably a debug build or something, build number is invalid
227          }
228          // These 2 need to happen *before* the launcher or the updater run, so we have valuable debug information, and so
229          // we can properly use websites that use Let's Encrypt (and other current certs not supported by old Java versions)
230          if(params.getLogLevel()==6){
231              TantalumConstants.setIsDebug(true);
232              Utils.getLogger().setLevel(Level.ALL);
233          } else{
234              TantalumConstants.setIsDebug(false);
235              Utils.getLogger().setLevel(Level.convert(params.getLogLevel()*100));
236          }
237  
238          Utils.getLogger().info("The Current Build Number"+ buildNumber.getBuildNumber());
239          //System.setProperty("dcm.config.filename","");
240          //runProxySetup(settings);
241          runStartupDebug(settings, params);
242          //injectNewRootCerts();
243          injectNewRootCerts(settings.getForceOverrideRootCerts() || params.isOverrideRoots() || ComputerInfo.isSchoolEnv());
244          //startLauncher(settings, params, directories, resources);
245          Relauncher launcher = new TechnicRelauncher(new HttpUpdateStream(), build, directories, resources, params);
246          if (params.isChangeOverride()){
247              settings.setAdvOptions(true);
248              settings.save();
249          }
250          try {
251              if (launcher.runAutoUpdater())
252                  startLauncher(settings, params, directories, resources);
253          } catch (InterruptedException e) {
254              //Canceled by user
255          } catch (DownloadException e) {
256              //
257          } catch (IOException e) {
258              e.printStackTrace();
259          }

264      public static void runProxySetup(@NotNull TantalumSettings settings2) {
265          if(settings2.getUseTorRelay()){
266              return;
267          }
268          if(settings2.getUseDNSOnly()) {
269              return;
270          }
271              //Setup http Proxy
272              if (settings2.getUseHTTPProxy()) {
273                  Utils.getLogger().info("HTTP Proxy is currently configured as \n  HTTP Proxy Host: " + settings2.getHTTPProxyHost() + "\n  HTTP Proxy Port: " + settings2.getHTTPProxyPort() + "\n HTTP Proxy Non Proxies Hosts: " + settings2.getHTTPProxyBypassDomains());
274                  System.setProperty("http.proxyHost", settings2.getHTTPProxyHost());
275                  System.setProperty("http.proxyPort", String.valueOf(settings2.getHTTPProxyPort()));
276                  System.setProperty("http.nonProxyHosts", settings2.getHTTPProxyBypassDomains());// Separated by |
277              }
278              //Setup Socks Proxy
279              if (settings2.getUseSocksProxy()) {
280                  Utils.getLogger().info("Socks Proxy is currently configured as \n  Socks Proxy Host: " + settings2.getSocksProxyHost() + "\n  Socks Proxy Port: " + settings2.getSocksProxyPort() + "\n Socks Proxy Using Version Five: " + settings2.getUseSocksProxyFive());
281                  System.setProperty("socksProxyHost", settings2.getSocksProxyHost());
282                  System.setProperty("socksProxyPort", String.valueOf(settings2.getSocksProxyPort()));
283                  if (!settings2.getUseSocksProxyFive()) {
284                      System.setProperty("socksProxyVersion", "4");
285                  }//Set Socks Version To Four if we aren't using v5
286              }
287

290      private static void checkIfRunningInsideOneDrive(File launcherRoot) {
291          if (OperatingSystem.getOperatingSystem() != OperatingSystem.WINDOWS) {
292              return;
293          }
294  
295          Path launcherRootPath = launcherRoot.toPath();
296  
297          for (String varName : new String[]{"OneDrive", "OneDriveConsumer"}) {
298              String varValue = System.getenv(varName);
299              if (varValue == null || varValue.isEmpty()) {
300                  continue;
301              }
302  
303              Path oneDrivePath = new File(varValue).toPath();
304  
305              if (launcherRootPath.startsWith(oneDrivePath)) {
306                  JOptionPane.showMessageDialog(null, "Technic Launcher cannot run inside OneDrive. Please move it out of OneDrive, in the launcher settings.", "Cannot run inside OneDrive", JOptionPane.ERROR_MESSAGE);
307              }
308          }
309      }

370      @SuppressWarnings("ConstantConditions")
371      private static void injectNewRootCerts(boolean forceInjection) {
372          // Adapted from Forge installer
373          final String javaVersion = System.getProperty("java.version");
374          if(forceInjection){Utils.getLogger().warning("Forcing New Root Certificates");}
375          if ((javaVersion == null || !javaVersion.startsWith("1.8.0_")) && !forceInjection) {
376              Utils.getLogger().log(Level.INFO, "Don't need to inject new root certificates: Java is newer than 8 (" + javaVersion + ")");
377              return;
378          }
379  
380          try {
381              if ((Integer.parseInt(javaVersion.substring("1.8.0_".length())) >= 101) && !forceInjection) {
382                  Utils.getLogger().log(Level.INFO, "Don't need to inject new root certificates: Java 8 is 101+ (" + javaVersion + ")");
383                  return;
384              }
385          } catch (final NumberFormatException e) {
386              Utils.getLogger().log(Level.WARNING,
387                      "Couldn't parse Java version, can't inject new root certs",
388                      e);
389              return;
390          }
391  
392          try {
393              KeyStore keyStore = KeyStore.getInstance(KeyStore.getDefaultType());
394              final Path ksPath = Paths.get(System.getProperty("java.home"),"lib", "security", "cacerts");
395              keyStore.load(Files.newInputStream(ksPath), "changeit".toCharArray());
396              Map<String, Certificate> jdkTrustStore = Collections.list(keyStore.aliases()).stream().collect(Collectors.toMap(a -> a, (String alias) -> {
397                  try {
398                      return keyStore.getCertificate(alias);
399                  } catch (KeyStoreException e) {
400                      Utils.getLogger().log(Level.WARNING, "Failed to get certificate", e);
401                      return null;
402                  }
403              }));
404  
405              KeyStore leKS = KeyStore.getInstance(KeyStore.getDefaultType());
406              InputStream leKSFile = LauncherMain.class.getResourceAsStream("/net/tharow/tantalum/launcher/resources/technickeystore.jks");
407              leKS.load(leKSFile, "technicrootca".toCharArray());
408              Map<String, Certificate> leTrustStore = Collections.list(leKS.aliases()).stream().collect(Collectors.toMap(a -> a, (String alias) -> {
409                  try {
410                      return leKS.getCertificate(alias);
411                  } catch (KeyStoreException e) {
412                      Utils.getLogger().log(Level.WARNING, "Failed to get certificate", e);
413                      return null;
414                  }
415              }));
416  
417              KeyStore mergedTrustStore = KeyStore.getInstance(KeyStore.getDefaultType());
418              mergedTrustStore.load(null, new char[0]);
419              for (final Map.Entry<String, Certificate> entry : jdkTrustStore.entrySet()) {
420                  mergedTrustStore.setCertificateEntry(entry.getKey(), entry.getValue());
421              }
422              for (final Map.Entry<String , Certificate> entry : leTrustStore.entrySet()) {
423                  mergedTrustStore.setCertificateEntry(entry.getKey(), entry.getValue());
424              }
425  
426              TrustManagerFactory instance = TrustManagerFactory.getInstance(TrustManagerFactory.getDefaultAlgorithm());
427              instance.init(mergedTrustStore);
428              SSLContext tls = SSLContext.getInstance("TLS");
429              tls.init(null, instance.getTrustManagers(), null);
430              HttpsURLConnection.setDefaultSSLSocketFactory(tls.getSocketFactory());
431              Utils.getLogger().log(Level.INFO, "Injected new root certificates");
432          } catch (KeyStoreException | IOException | NoSuchAlgorithmException | CertificateException | KeyManagementException e) {
433              Utils.getLogger().log(Level.WARNING, "Failed to inject new root certificates. Problems might happen");
434              e.printStackTrace();
435          }
436      }

441          System.setProperty("xr.load.xml-reader", "org.ccil.cowan.tagsoup.Parser");
442  
443          //DnsCacheManipulator.clearDnsCache();
444  
445          //DnsCacheManipulator.loadDnsCacheConfig("net.tharow.tantalum.resources.dns-cache.properties");
446  
447          //setupDnsJava(directories);
448  
449          // runProxySetup(settings, directories.getTorRelayDirectory()); //Check and run proxy and custom dns settings
450  
451          //Remove all log files older than a week
452          new Thread(() -> {
453              Iterator<File> files = FileUtils.iterateFiles(new File(directories.getLauncherDirectory(), "logs"), new String[] {"log"}, false);
454              while (files.hasNext()) {
455                  File logFile = files.next();
456                  if (logFile.exists() && (new DateTime(logFile.lastModified())).isBefore(DateTime.now().minusWeeks(1)))
457                      //noinspection ResultOfMethodCallIgnored
458                      logFile.delete();
459              }
460          }).start();
461  
462          final SplashScreen splash = new SplashScreen(resources.getImage("launch_splash.png"), 0);

464          splash.getContentPane().setBackground(new Color (bg.getRed(),bg.getGreen(),bg.getBlue(),255));
465          splash.pack();
466          splash.setLocationRelativeTo(null);
467          splash.setVisible(true);
468  
469          JavaVersionRepository javaVersions = new JavaVersionRepository();
470          (new InstalledJavaSource()).enumerateVersions(javaVersions);
471          FileJavaSource javaVersionFile = FileJavaSource.load(new File(settings.getTechnicRoot(), "javaVersions.json"));
472          javaVersionFile.enumerateVersions(javaVersions);
473          javaVersions.selectVersion(settings.getJavaVersion(), settings.getJavaBitness());
474          TantalumUserStore users = TantalumUserStore.load(new File(directories.getLauncherDirectory(), "users.json"));
475          UserModel userModel = new UserModel(users, new MojangAuthenticator(users.getClientToken(), "https://blessingskin.azurewebsites.net/api/yggdrasil/authserver/"));
476          if (TantalumConstants.isUnlocked()) {
477              MojangAuthenticator mojangAuthenticator = new MojangAuthenticator(users.getClientToken());
