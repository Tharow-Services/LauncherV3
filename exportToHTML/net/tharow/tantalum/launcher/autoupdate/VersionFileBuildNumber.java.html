19   package net.tharow.tantalum.launcher.autoupdate;
20   
21   import net.tharow.tantalum.autoupdate.IBuildNumber;
22   import net.tharow.tantalum.ui.lang.ResourceLoader;
23   import org.apache.commons.io.IOUtils;
24   
25   import java.nio.charset.StandardCharsets;
26   
27   public class VersionFileBuildNumber implements IBuildNumber {
28       private final ResourceLoader resources;
29   
30       public VersionFileBuildNumber(ResourceLoader resources) {
31           this.resources = resources;
32       }
33   
34       @Override
35       public String getBuildNumber() {
36           String build = "0";
37           try {
38               build = IOUtils.toString(resources.getResourceAsStream("/version"), StandardCharsets.UTF_8);
39           } catch (Exception e) {
40               e.printStackTrace();
41           }
42   
43           return build.substring(0,3);
44       }
45   }
