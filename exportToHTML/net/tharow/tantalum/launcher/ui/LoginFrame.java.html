149      @Override
150      public void relocalize(ResourceLoader loader) {
151          this.resources = loader;
152          this.resources.registerResource(this);
153  
154          setIconImage(this.resources.getImage("icon.png"));
155  
156          //Wipe controls
157          this.getContentPane().removeAll();
158          this.setLayout(null);
159  
160          //Clear references to existing controls
161          nameSelect = null;
162          rememberAccount = null;
163          password = null;
164  
165          initComponents();
166  
167          refreshSelectedUsers();
168  
169          EventQueue.invokeLater(() -> {
170              invalidate();
171              repaint();
172          });
173      }

175      @Override
176      public void userChanged(IUserType user) {
177          if (user == null) {
178              this.setVisible(true);
179              refreshSelectedUsers();
180  
181              if (nameSelect.isVisible())
182                  nameSelect.grabFocus();
183              else
184                  username.grabFocus();
185  
186              EventQueue.invokeLater(() -> {
187                  invalidate();
188                  repaint();
189              });
190          } else
191              this.setVisible(false);
192      }

513      protected void refreshSelectedUsers() {
514          Collection<IUserType> users = userModel.getUsers();
515          IUserType lastUser = userModel.getLastUser();
516  
517          if (users.size() == 0) {
518              setAccountSelectVisibility(false);
519              setAddAccountVisibility(true);
520              login.setVisible(false);
521              clearCurrentUser();
522          } else {
523              nameSelect.setVisible(true);
524              nameSelect.removeAllItems();
525  
526              for (IUserType account : users) {
527                  nameSelect.addItem(account);
528              }
529  
530              if (lastUser == null)
531                  lastUser = users.iterator().next();
532  
533              setCurrentUser(lastUser);
534          }
535      }

537      protected void login() {
538          if (username.isVisible()) {
539              newMojangLogin(username.getText());
540              return;
541          }
542          if (username2.isVisible()) {
543              newAuthlibLogin(username2.getText(),authServer.getText());
544              return;
545          }
546  
547          IUserType user = (IUserType) nameSelect.getSelectedItem();
548  
549          // Can't login with a null user. This should never be hit.
550          if (user == null) {
551              Utils.getLogger().log(Level.WARNING, "Attempted to login with a null user.");
552              return;
553          }
554  
555          login(user);
556      }

642      private void login(IUserType user) {
643          try {
644              user.login(userModel);
645              userModel.addUser(user);
646              userModel.setCurrentUser(user);
647              setCurrentUser(user);
648          } catch (SessionException e) {
649              showMessageDialog(
650                      this, "Please log in again for user " + user.getDisplayName(),
651                      e.getMessage(), ERROR_MESSAGE);
652              forgetUser(user);
653          } catch (ResponseException e) {
654              showMessageDialog(
655                      this, e.getMessage(), e.getError(), ERROR_MESSAGE);
656              forgetUser(user);
657          } catch (AuthenticationException e) {
658              Utils.getLogger().log(Level.SEVERE, e.getMessage(), e);
659  
660              //Couldn't reach auth server- if we're running silently (we just started up and have a user session ready to roll)
661              //Go ahead and just play offline automatically, like the minecraft client does.  If we're running loud (user
662              //is actually at the login UI clicking the login button), give them a choice.
663              if (JOptionPane.YES_OPTION == JOptionPane.showConfirmDialog(this,
664                      "The auth servers at The Tantalum Platform are inaccessible.  Would you like to play offline?",
665                      "Offline Play", JOptionPane.YES_NO_OPTION, JOptionPane.WARNING_MESSAGE)) {
666  
667                  // This is the last time we'll have access to the user's real username,
668                  // so we should set the last-used username now
669                  userModel.setLastUser(user);
670                  userModel.setCurrentUser(userModel.getMojangAuthenticator().createOfflineUser(user.getDisplayName()));
671                  setCurrentUser(userModel.getCurrentUser());
672              }
673          }
674      }

681      private void newMicrosoftLogin() {
682          setAccountSelectVisibility(false);
683          setAddAccountVisibility(false);
684          // TODO: Setup info message + cancel flow
685  
686          try {
687              MicrosoftUser microsoftUser = userModel.getMicrosoftAuthenticator().loginNewUser();
688              userModel.addUser(microsoftUser);
689              userModel.setCurrentUser(microsoftUser);
690              setCurrentUser(microsoftUser);
691          } catch (MicrosoftAuthException e) {
692              switch (e.getType()) {
693                  case UNDERAGE : showMessage("microsoft.underage"); break;
694                  case NO_XBOX_ACCOUNT : {
695                      showMessage("microsoft.noxbox");
696                      DesktopUtils.browseUrl("https://www.minecraft.net/login");
697                      break;
698                  }
699                  case NO_MINECRAFT : showMessage("microsoft.nomc"); break;
700                  default : {
701                      e.printStackTrace();
702                      showMessageDialog(this, e.getMessage(), resources.getString("login.microsoft.default.title"), ERROR_MESSAGE);
703                  }
704              }
705          } finally {
706              setAccountSelectVisibility(!userModel.getUsers().isEmpty());
707              setAddAccountVisibility(true);
708          }
709      }
