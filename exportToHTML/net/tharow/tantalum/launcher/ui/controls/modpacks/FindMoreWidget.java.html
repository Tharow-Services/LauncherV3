19   package net.tharow.tantalum.launcher.ui.controls.modpacks;
20   
21   import net.tharow.tantalum.launcher.ui.LauncherFrame;
22   import net.tharow.tantalum.launcher.ui.controls.SelectorWidget;
23   import net.tharow.tantalum.ui.lang.ResourceLoader;
24   
25   import javax.swing.*;
26   import javax.swing.border.EmptyBorder;
27   import java.awt.*;
28   
29   public class FindMoreWidget extends SelectorWidget {
30       private final JLabel moreLabel;
31   
32       public FindMoreWidget(ResourceLoader resources) {
33           super(resources);
34   
35           initComponents();
36           setIsSelected(true);
37   
38           setLayout(new GridBagLayout());
39           setBorder(new EmptyBorder(5,0,5,0));
40           moreLabel = new JLabel(resources.getString("launcher.packselector.more"), JLabel.CENTER);
41           moreLabel.setFont(getResources().getFont(ResourceLoader.FONT_OPENSANS, 14));
42           moreLabel.setForeground(LauncherFrame.COLOR_DIM_TEXT);
43           moreLabel.setIcon(resources.getIcon("arrow_right.png"));
44           moreLabel.setHorizontalTextPosition(SwingConstants.LEADING);
45           moreLabel.setIconTextGap(8);
46           moreLabel.setOpaque(false);
47           add(moreLabel, new GridBagConstraints(0, 0, 1, 1, 1, 1, GridBagConstraints.CENTER, GridBagConstraints.BOTH, new Insets(0, 0, 0, 0), 0, 0));
48       }
49   
50       public void setWidgetData(String text) {
51           moreLabel.setText(text);
52       }
53   
54       public String getWidgetData() {
55           return moreLabel.getText();
56       }
57   }
