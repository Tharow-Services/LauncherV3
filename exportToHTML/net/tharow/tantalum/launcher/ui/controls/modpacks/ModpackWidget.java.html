19   package net.tharow.tantalum.launcher.ui.controls.modpacks;
20   
21   import net.tharow.tantalum.ui.lang.ResourceLoader;
22   import net.tharow.tantalum.launcher.ui.LauncherFrame;
23   import net.tharow.tantalum.launcher.ui.controls.SelectorWidget;
24   import net.tharow.tantalum.launchercore.image.IImageJobListener;
25   import net.tharow.tantalum.launchercore.image.ImageJob;
26   import net.tharow.tantalum.launchercore.modpacks.ModpackModel;
27   import net.tharow.tantalum.utilslib.ImageUtils;
28   
29   import javax.swing.*;
30   import javax.swing.border.LineBorder;
31   import java.awt.*;
32   
33   public class ModpackWidget extends SelectorWidget implements IImageJobListener<ModpackModel> {
34       private final ModpackModel modpack;
35       private final ImageJob<ModpackModel> imageJob;
36       private final ResourceLoader resources;
37   
38       private JLabel icon;
39       private JLabel displayName;
40       private JLabel updateIcon;
41   
42       public ModpackWidget(ResourceLoader resources, ModpackModel modpack, ImageJob<ModpackModel> job) {
43           super(resources);
44   
45           this.resources = resources;
46           this.imageJob = job;
47           imageJob.addJobListener(this);
48           this.modpack = modpack;
49           initComponents();
50       }
51   
52       public ModpackModel getModpack() {
53           return modpack;
54       }
55   
56       protected void initComponents() {
57           super.initComponents();
58           setBorder(BorderFactory.createEmptyBorder(4,20,4,8));
59           setLayout(new GridBagLayout());
60   
61           icon = new JLabel();
62           icon.setIcon(new ImageIcon(ImageUtils.scaleWithAspectWidth(imageJob.getImage(), 32)));
63           add(icon, new GridBagConstraints(0,0,1,1,0,0,GridBagConstraints.WEST, GridBagConstraints.NONE, new Insets(0,0,0,14),0,0));
64   
65           displayName = new JLabel(modpack.getDisplayName());
66           displayName.setFont(getResources().getFont(ResourceLoader.FONT_OPENSANS, 14));
67           displayName.setForeground(LauncherFrame.COLOR_WHITE_TEXT);
68           displayName.setMaximumSize(new Dimension(210, displayName.getPreferredSize().height));
69           add(displayName, new GridBagConstraints(1,0,1,1,1,0,GridBagConstraints.WEST, GridBagConstraints.BOTH, new Insets(0,0,0,0),0,0));
70           updateIcon = new JLabel();
71           updateIcon.setIcon(getResources().getIcon("update_available.png"));
72           add(updateIcon, new GridBagConstraints(2,0,1,1,0,0,GridBagConstraints.WEST, GridBagConstraints.NONE, new Insets(0,0,0,5),0,0));
73           updateIcon.setVisible(modpack.hasRecommendedUpdate());
74       }
75   
76       public void updateFromPack(ImageJob<ModpackModel> job) {
77           displayName.setText(modpack.getDisplayName());
78           updateIcon.setVisible(modpack.hasRecommendedUpdate());
79           icon.setIcon(new ImageIcon(ImageUtils.scaleWithAspectWidth(job.getImage(), 32)));
80           job.addJobListener(this);
81       }
82   
83       @Override
84       public void jobComplete(ImageJob<ModpackModel> job) {
85           icon.setIcon(new ImageIcon(ImageUtils.scaleWithAspectWidth(job.getImage(), 32)));
86           revalidate();
87       }
88   
89       @Override
90       public JToolTip createToolTip() {
91           JToolTip toolTip = new JToolTip();
92           toolTip.setBackground(LauncherFrame.COLOR_FOOTER);
93           toolTip.setForeground(LauncherFrame.COLOR_GREY_TEXT);
94           toolTip.setBorder(BorderFactory.createCompoundBorder(new LineBorder(LauncherFrame.COLOR_GREY_TEXT), BorderFactory.createEmptyBorder(5,5,5,5)));
95           toolTip.setFont(resources.getFont(ResourceLoader.FONT_OPENSANS, 14));
96   
97           return toolTip;
98       }
99   }
