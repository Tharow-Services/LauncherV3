94       private final DocumentFilter documentFilter = new DocumentFilter(){
95           @Override
96           public void insertString(FilterBypass fb, int offset, String string, AttributeSet attr) throws BadLocationException
97           {
98               if(fb.getDocument().getLength() + string.length() <= MAX_SEARCH_STRING)
99               {
100                  fb.insertString(offset, string, attr);
101              }
102          }
103  
104          @Override
105          public void remove(FilterBypass fb, int offset, int length) throws BadLocationException
106          {
107              fb.remove(offset, length);
108          }
109  
110          @Override
111          public void replace(FilterBypass fb, int offset, int length, String text, AttributeSet attrs)throws BadLocationException
112          {
113              int finalTextLength = (fb.getDocument().getLength() - length) + text.length();
114              if (finalTextLength > MAX_SEARCH_STRING)
115                  text = text.substring(0, text.length() - (finalTextLength-MAX_SEARCH_STRING));
116              fb.replace(offset, length, text, attrs);
117          }
118      };

273      protected void addModpackInternal(ModpackModel modpack) {
274          final ModpackWidget widget = new ModpackWidget(resources, modpack, iconRepo.startImageJob(modpack));
275  
276          if (modpack.hasRecommendedUpdate()) {
277              widget.setToolTipText(resources.getString("launcher.packselector.updatetip"));
278          }
279          widget.addActionListener(e -> {
280              if (e.getSource() instanceof ModpackWidget)
281                  selectWidget((ModpackWidget)e.getSource());
282          });
283  
284          if (widget.getModpack().isSelected()) {
285              selectedWidget = widget;
286          }
287  
288          if (allModpacks.containsKey(modpack.getName()) && allModpacks.get(modpack.getName()).isSelected())
289              selectedWidget = widget;
290  
291          allModpacks.put(modpack.getName(), widget);
292  
293          rebuildUI();
294  
295          if (selectedWidget != null) {
296              EventQueue.invokeLater(() -> {
297                  if (widget == selectedWidget)
298                      selectWidget(widget);
299                  else
300                      selectedWidget.scrollRectToVisible(new Rectangle(selectedWidget.getSize()));
301              });
302          }
303      }

305      @Override
306      public void refreshComplete() {
307          setTintActive(false);
308          //Debug.getConfig(defaultPacks);
309          if (findMoreWidget.getWidgetData().equals(resources.getString("launcher.packselector.api"))) {
310              if (allModpacks.size() == 0) {
311                  findMoreWidget.setWidgetData(resources.getString("launcher.packselector.badapi"));
312                  findMoreUrl = "https://www.technicpack.net/";
313              } else {
314                  for(ModpackWidget widget : allModpacks.values()) {
315                      findMoreUrl = widget.getModpack().getWebSite();
316                      break;
317                  }
318              }
319          }
320  
321          if (selectedWidget == null || selectedWidget.getModpack() == null || !allModpacks.containsKey(selectedWidget.getModpack().getName())) {
322              java.util.List<ModpackWidget> sortedPacks = new LinkedList<>(allModpacks.values());
323              sortedPacks.sort((o1, o2) -> {
324                  int priorityCompare = Integer.compare(o2.getModpack().getPriority(), o1.getModpack().getPriority());
325                  if (priorityCompare != 0)
326                      return priorityCompare;
327                  else if (o1.getModpack().getDisplayName() == null && o2.getModpack().getDisplayName() == null)
328                      return 0;
329                  else if (o1.getModpack().getDisplayName() == null)
330                      return -1;
331                  else if (o2.getModpack().getDisplayName() == null)
332                      return 1;
333                  else
334                      return o1.getModpack().getDisplayName().compareToIgnoreCase(o2.getModpack().getDisplayName());
335              });
336              if (sortedPacks.size() > 0) {
337                  selectWidget(sortedPacks.get(0));
338              }
339          }
340      }

342      protected void selectWidget(ModpackWidget widget) {
343          if (selectedWidget != null)
344              selectedWidget.setIsSelected(false);
345          selectedWidget = widget;
346          selectedWidget.setIsSelected(true);
347          selectedWidget.getModpack().select();
348          selectedWidget.scrollRectToVisible(new Rectangle(selectedWidget.getSize()));
349  
350          if (modpackInfoPanel != null)
351              modpackInfoPanel.setModpack(widget.getModpack());
352  
353              final ModpackWidget refreshWidget = selectedWidget;
354              Thread thread = new Thread("Modpack redownload " + selectedWidget.getModpack().getDisplayName()) {
355                  @Override
356                  public void run() {
357                      try {
358                          PlatformPackInfo updatedInfo = tantalum.getPlatformPackInfo(refreshWidget.getModpack().getName());
359                          PackInfo infoToUse = updatedInfo;
360  
361                          if (updatedInfo != null && updatedInfo.hasSolder()) {
362                              try {
363                                  ISolderPackApi solderPack = solderApi.getSolderPack(updatedInfo.getSolder(), updatedInfo.getName(), solderApi.getMirrorUrl(updatedInfo.getSolder()));
364                                  infoToUse = new CombinedPackInfo(solderPack.getPackInfo(), updatedInfo);
365                              } catch (RestfulAPIException ignored) {
366                              }
367                          }
368  
369                          if (infoToUse != null)
370                              refreshWidget.getModpack().setPackInfo(infoToUse);
371  
372                          EventQueue.invokeLater(() -> {
373                              if (modpackInfoPanel != null)
374                                  modpackInfoPanel.setModpackIfSame(refreshWidget.getModpack());
375  
376                              if (refreshWidget.getModpack().hasRecommendedUpdate()) {
377                                  refreshWidget.setToolTipText(resources.getString("launcher.packselector.updatetip"));
378                              } else {
379                                  refreshWidget.setToolTipText(null);
380                              }
381  
382                              iconRepo.refreshRetry(refreshWidget.getModpack());
383                              refreshWidget.updateFromPack(iconRepo.startImageJob(refreshWidget.getModpack()));
384  
385                              EventQueue.invokeLater(() -> {
386                                  revalidate();
387                                  repaint();
388                              });
389                          });
390  
391                      } catch (RestfulAPIException ex) {
392                          ex.printStackTrace();
393                      }
394                  }
395              };
396              thread.start();
397      }

399      protected void rebuildUI() {
400          if (!EventQueue.isDispatchThread()) {
401              EventQueue.invokeLater(this::rebuildUI);
402              return;
403          }
404  
405          GridBagConstraints constraints = new GridBagConstraints(0, 0, 1, 1, 1.0, 0.0, GridBagConstraints.WEST, GridBagConstraints.HORIZONTAL, new Insets(0,0,0,0), 0,0);
406  
407          java.util.List<ModpackWidget> sortedPacks = new LinkedList<>(allModpacks.values());
408          sortedPacks.sort((o1, o2) -> {
409              int priorityCompare = Integer.compare(o2.getModpack().getPriority(), o1.getModpack().getPriority());
410              if (priorityCompare != 0)
411                  return priorityCompare;
412              else if (o1.getModpack().getDisplayName() == null && o2.getModpack().getDisplayName() == null)
413                  return 0;
414              else if (o1.getModpack().getDisplayName() == null)
415                  return -1;
416              else if (o2.getModpack().getDisplayName() == null)
417                  return 1;
418              else
419                  return o1.getModpack().getDisplayName().compareToIgnoreCase(o2.getModpack().getDisplayName());
420          });
421  
422          widgetList.removeAll();
423  
424          for(ModpackWidget sortedPack : sortedPacks) {
425              widgetList.add(sortedPack, constraints);
426              constraints.gridy++;
427          }
428  
429          if (filterContents.getText().length() >= 3) {
430              widgetList.add(findMoreWidget, constraints);
431          }
432  
433          widgetList.add(Box.createHorizontalStrut(294), constraints);
434          constraints.gridy++;
435  
436          constraints.weighty = 1.0;
437          widgetList.add(Box.createGlue(), constraints);
438  
439          EventQueue.invokeLater(() -> {
440              revalidate();
441              repaint();
442          });
443      }

477      protected void detectFilterChanges() {
478          cancelJob();
479  
480          if (filterContents.getText().length() >= 3) {
481              loadNewJob(filterContents.getText());
482          } else if (lastFilterContents.length() >= 3) {
483              clear();
484              defaultPacks.addPassthroughContainer(this);
485              for(ModpackModel modpack : defaultPacks.getModpacks()) {
486                  addModpackToContainer(modpack);
487              }
488              refreshComplete();
489          }
490  
491          lastFilterContents = filterContents.getText();
492      }
