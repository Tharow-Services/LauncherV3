23   import org.xhtmlrenderer.swing.ImageResourceLoader;
24   import org.xhtmlrenderer.swing.MutableFSImage;
25   
26   /**
27    * Denotes a URI pointing to an image that needs to be loaded. BackgroundImageLoaderItem are queued inside a
28    * BackgroundImageLoader and loaded by a BackgroundImageLoaderThread.
29    */
30   class ImageLoadItem {
31       final ImageResourceLoader _imageResourceLoader;
32       final String _uri;
33       final MutableFSImage _mfsImage;
34       final int _targetWidth;
35       final int _targetHeight;
36   
37       public ImageLoadItem(final ImageResourceLoader imageResourceLoader, String uri, MutableFSImage fsi, int width, int height) {
38           this._imageResourceLoader = imageResourceLoader;
39           this._uri = uri;
40           this._mfsImage = fsi;
41           this._targetWidth = width;
42           this._targetHeight = height;
43       }
44   
45       public boolean haveTargetDimensions() {
46           return _targetWidth > -1 && _targetHeight > -1;
47       }
48   }
