70       public void refresh(ModpackModel model) {

81       protected void selectRecommended() {
82           manualBuildList.setBorder(new RoundBorder(LauncherFrame.COLOR_GREY_TEXT, 1, 10));
83           manualBuildList.setEnabled(false);
84           manualBuildList.setSelectedItem(new PackBuildItem(modpack.getRecommendedBuild(), resources, modpack));
85           this.modpack.setBuild(InstalledPack.RECOMMENDED);
86       }
87   
88       protected void selectLatest() {
89           manualBuildList.setBorder(new RoundBorder(LauncherFrame.COLOR_GREY_TEXT, 1, 10));
90           manualBuildList.setEnabled(false);
91           manualBuildList.setSelectedItem(new PackBuildItem(modpack.getLatestBuild(), resources, modpack));
92           this.modpack.setBuild(InstalledPack.LATEST);
93       }
94   
95       protected void selectManual() {
96           if (manualBuildList.getItemCount() == 0)
97               return;
98   
99           // Nothing is selected, try to set it to the currently installed modpack build
100          // (could be an actual build string (like "1.0"), "recommended" or "latest")
101          if (manualBuildList.getSelectedItem() == null)
102              manualBuildList.setSelectedItem(new PackBuildItem(modpack.getBuild(), resources, modpack));
103  
104          // Still nothing is selected, try to fallback to the recommended build
105          if (manualBuildList.getSelectedItem() == null)
106              manualBuildList.setSelectedItem(new PackBuildItem(modpack.getRecommendedBuild(), resources, modpack));
107  
108          // Still nothing is selected, and both the currently installed and recommended builds couldn't be selected,
109          // so we try to select the first available one (we already know there's at least 1 build available)
110          if (manualBuildList.getSelectedIndex() == -1) {
111              manualBuildList.setSelectedIndex(0);
112          }
113  
114          this.modpack.setBuild(((PackBuildItem) manualBuildList.getSelectedItem()).getBuildNumber());
115          manualBuildList.setBorder(new RoundBorder(LauncherFrame.COLOR_BUTTON_BLUE, 1, 10));
116          manualBuildList.setEnabled(true);
117      }
118  
119      protected void buildUpdated() {
120          this.modpack.setBuild(((PackBuildItem)manualBuildList.getSelectedItem()).getBuildNumber());
121      }
122  
123      protected void deletePack() {
124          int result = JOptionPane.showConfirmDialog(this, resources.getString("modpackoptions.delete.confirmtext"), resources.getString("modpackoptions.delete.confirmtitle"), JOptionPane.YES_NO_OPTION, JOptionPane.WARNING_MESSAGE);
125          if (result == JOptionPane.YES_OPTION) {
126              modpack.delete();
127              dispose();
128          }
129      }
130  
131      protected void resetPack() {
132          int result = JOptionPane.showConfirmDialog(this, resources.getString("modpackoptions.reinstall.confirmtext"), resources.getString("modpackoptions.reinstall.confirmtitle"), JOptionPane.YES_NO_OPTION, JOptionPane.WARNING_MESSAGE);
133          if (result == JOptionPane.YES_OPTION) {
134              modpack.resetPack();
135              refresh(modpack);
136          }
137      }
138  
139      protected void openFolder() {
140          if (modpack.getInstalledDirectory() != null && modpack.getInstalledDirectory().exists())
141              DesktopUtils.open(modpack.getInstalledDirectory());
142      }
143  
144      protected void moveFolder() {
145          int result = chooser.showOpenDialog(this);
146  
147          if (result == JFileChooser.APPROVE_OPTION) {
148              File file = chooser.getSelectedFile();
149  
150              if (file.list().length > 0) {
151                  JOptionPane.showMessageDialog(this, resources.getString("modpackoptions.move.errortext"), resources.getString("modpackoptions.move.errortitle"), JOptionPane.WARNING_MESSAGE);
152                  return;
153              }
154  
155              modpack.setInstalledDirectory(file);
156              refresh(modpack);
157          }
158      }

355      private void initValues() {
356          File installDir = modpack.getInstalledDirectory();
357          if (installDir == null) {
358              installField.setText(resources.getString("modpackoptions.installfolder.none"));
359          } else {
360              installField.setText(installDir.getAbsolutePath());
361          }
362  
363          for (String build : modpack.getBuilds()) {
364              //noinspection unchecked
365              manualBuildList.insertItemAt(new PackBuildItem(build, resources, modpack), 0);
366          }
367  
368          if (manualBuildList.getItemCount() == 0) {
369              recommended.setEnabled(false);
370              latest.setEnabled(false);
371              manual.setEnabled(false);
372              recommended.setSelected(true);
373              //noinspection unchecked
374              manualBuildList.addItem(new PackBuildItem(resources.getString("modpackoptions.version.missing"), resources, modpack));
375              manualBuildList.setEnabled(false);
376              return;
377          }
378  
379          String build = modpack.getBuild();
380  
381          if (build.equals(InstalledPack.RECOMMENDED)) {
382              recommended.setSelected(true);
383              selectRecommended();
384          } else if (build.equals(InstalledPack.LATEST)) {
385              latest.setSelected(true);
386              selectLatest();
387          } else {
388              manual.setSelected(true);
389              selectManual();
390          }
391      }
