19   package net.tharow.tantalum.launcher.ui.components.news;

22   import net.tharow.tantalum.ui.lang.ResourceLoader;
23   import net.tharow.tantalum.launcher.ui.LauncherFrame;
24   import net.tharow.tantalum.ui.controls.RoundedButton;
25   import net.tharow.tantalum.ui.controls.list.SimpleScrollbarUI;

28   import net.tharow.tantalum.platform.io.NewsArticle;

30   
31   import javax.swing.*;
32   import javax.swing.event.HyperlinkEvent;
33   import java.awt.*;
34   import java.awt.event.*;
35   import java.beans.PropertyChangeEvent;
36   import java.beans.PropertyChangeListener;
37   
38   public class NewsInfoPanel extends JPanel implements PropertyChangeListener {
39       private final ResourceLoader resources;
40       private final ImageRepository<AuthorshipInfo> avatarRepo;
41   
42       JTextPane newsText;
43       JScrollPane newsScroller;
44       AuthorshipWidget authorshipInfo;
45       JLabel title;
46   
47       private String url = "";
48   
49       public NewsInfoPanel(ResourceLoader resources, ImageRepository<AuthorshipInfo> avatarRepo) {
50   
51           this.resources = resources;
52           this.avatarRepo = avatarRepo;
53   
54           initComponents();
55       }
56   
57       public void setArticle(NewsArticle article) {
58           if (article == null) {
59               newsText.setText("");
60               url = "";
61               return;
62           }
63   
64           title.setText(article.getTitle());
65           url = article.getUrl();
66           newsText.setText("<html><head><style type=\"text/css\">a{color:#309aeb}body{font-family:\"" + newsText.getFont().getFamily() + "\";color:#D0D0D0}</style></head><body>"+
67               article.getContent() +
68               "</body></html>");
69   
70           //noinspection unchecked
71           authorshipInfo.setAuthorshipInfo(article.getAuthorshipInfo(), avatarRepo.startImageJob(article.getAuthorshipInfo()));
72   
73           EventQueue.invokeLater(() -> newsText.scrollRectToVisible(new Rectangle(0, 0, 1, 1)));
74       }
75   
76       protected void visitCurrentItem() {
77           if (url != null && !url.equals("")) {
78               DesktopUtils.browseUrl(url);
79           }
80       }
81   
82       private void initComponents() {
83           setLayout(new GridBagLayout());
84           setBorder(BorderFactory.createEmptyBorder(20,20,18,16));
85           setBackground(LauncherFrame.COLOR_CENTRAL_BACK_OPAQUE);
86   
87           title = new JLabel("");
88           title.setForeground(LauncherFrame.COLOR_WHITE_TEXT);
89           title.setFont(resources.getFont(ResourceLoader.FONT_RALEWAY, 26));
90           title.setCursor(Cursor.getPredefinedCursor(Cursor.HAND_CURSOR));
91   
92           title.addMouseListener(new SimpleMouseListener(this::visitCurrentItem));
93   
94           add(title, new GridBagConstraints(0, 0, 2, 1, 1.0, 0.0, GridBagConstraints.NORTHWEST, GridBagConstraints.HORIZONTAL, new Insets(0, 0, 0, 0), 0, 0));
95   
96           authorshipInfo = new AuthorshipWidget(resources);
97           add(authorshipInfo, new GridBagConstraints(0, 1, 2, 1, 1.0, 0.0, GridBagConstraints.NORTHWEST, GridBagConstraints.NONE, new Insets(0, 0, 8, 0), 0, 0));
98   
99           newsText = new JTextPane();
100          newsText.setBorder(BorderFactory.createEmptyBorder(5, 5, 5, 5));
101          newsText.setOpaque(false);
102          newsText.setForeground(LauncherFrame.COLOR_WHITE_TEXT);
103          newsText.setFont(resources.getFont(ResourceLoader.FONT_OPENSANS, 16));
104          newsText.setEditable(false);
105          newsText.setHighlighter(null);
106          newsText.setAlignmentX(LEFT_ALIGNMENT);
107          newsText.setContentType("text/html");
108          newsText.addHyperlinkListener(e -> {
109              if (e.getEventType() == HyperlinkEvent.EventType.ACTIVATED) {
110                  if (e.getURL() != null)
111                      DesktopUtils.browseUrl(e.getURL().toString());
112              }
113          });
114          newsText.addComponentListener(new ComponentListener() {
115              @Override
116              public void componentResized(ComponentEvent e) {
117                  NewsInfoPanel.this.revalidate();
118              }
119  
120              @Override
121              public void componentMoved(ComponentEvent e) {
122  
123              }
124  
125              @Override
126              public void componentShown(ComponentEvent e) {
127  
128              }
129  
130              @Override
131              public void componentHidden(ComponentEvent e) {
132  
133              }
134          });
135  
136          newsScroller = new JScrollPane(newsText, JScrollPane.VERTICAL_SCROLLBAR_AS_NEEDED, JScrollPane.HORIZONTAL_SCROLLBAR_NEVER);
137          newsScroller.getVerticalScrollBar().setUI(new SimpleScrollbarUI(LauncherFrame.COLOR_SCROLL_TRACK, LauncherFrame.COLOR_SCROLL_THUMB));
138          newsScroller.getVerticalScrollBar().setPreferredSize(new Dimension(10, 10));
139          newsScroller.setBorder(BorderFactory.createEmptyBorder());
140          newsScroller.setMaximumSize(new Dimension(32000,900));
141          newsScroller.setOpaque(false);
142          newsScroller.getViewport().setOpaque(false);
143  
144          JPanel newsTextPanel = new JPanel();
145          newsTextPanel.setLayout(new BoxLayout(newsTextPanel, BoxLayout.PAGE_AXIS));
146          newsTextPanel.setOpaque(false);
147          newsTextPanel.add(newsScroller);
148          newsTextPanel.add(Box.createVerticalGlue());
149  
150          add(newsTextPanel, new GridBagConstraints(0, 2, 2, 1, 1.0, 1.0, GridBagConstraints.NORTHWEST, GridBagConstraints.BOTH, new Insets(0, 0, 10, 0), 0, 0));
151  
152          add(Box.createGlue(), new GridBagConstraints(0, 3, 1, 1, 1.0, 0.0, GridBagConstraints.NORTHWEST, GridBagConstraints.NONE, new Insets(0, 0, 0, 0), 0, 0));
153  
154          RoundedButton discussButton = new RoundedButton(resources.getString("launcher.news.discuss"));
155          discussButton.setFont(resources.getFont(ResourceLoader.FONT_OPENSANS, 16));
156          discussButton.setBorder(BorderFactory.createEmptyBorder(5, 17, 10, 17));
157          discussButton.setForeground(LauncherFrame.COLOR_BUTTON_BLUE);
158          discussButton.setHoverForeground(LauncherFrame.COLOR_BLUE);
159          discussButton.setAlignmentX(RIGHT_ALIGNMENT);
160          discussButton.setContentAreaFilled(false);
161          discussButton.addActionListener(e -> visitCurrentItem());
162          add(discussButton, new GridBagConstraints(1, 3, 1, 1, 0.0, 0.0, GridBagConstraints.SOUTHEAST, GridBagConstraints.BOTH, new Insets(0, 0, 0, 0), 0, 0));
163      }
164  
165      @Override
166      public void propertyChange(PropertyChangeEvent evt) {
167          invalidate();
168      }
169  }
