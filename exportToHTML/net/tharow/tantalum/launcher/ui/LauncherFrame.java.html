193      private boolean TestNews() {
194          AtomicBoolean temp = new AtomicBoolean(false);
195          Thread thread = new Thread(() -> {
196              try {
197                  RepoReleasesData.getRestObject(TantalumConstants.NEWS_URL);
198                  temp.set(true);
199              } catch (RestfulAPIException e) {
200                  Utils.getLogger().log(Level.WARNING, "Unable to load news", e);
201              }
202          });
203          thread.start();
204          return temp.get();
205      }

248      protected void launchModpack() {
249          ModpackModel pack = modpackSelector.getSelectedPack();
250          boolean requiresInstall = false;
251  
252          if (pack == null || (pack.getInstalledPack() == null && (pack.getPackInfo() == null || !pack.getPackInfo().isComplete())))
253              return;
254  
255          if (pack.getInstalledDirectory() == null) {
256              requiresInstall = true;
257              pack.save();
258              modpackSelector.forceRefresh();
259          }
260  
261          boolean forceInstall = false;
262          Version installedVersion = pack.getInstalledVersion();
263  
264          //Force a full install (check cache, redownload, unzip files) if we have no current installation of this modpack
265          if (installedVersion == null) {
266              forceInstall = true;
267              requiresInstall = true;
268          } else if (pack.getBuild() != null && !pack.isLocalOnly()) {
269  
270              //Ask the user if they want to update to the newer version if:
271              //1- the pack build is RECOMMENDED & the recommended version is diff from the installed version
272              //2- the pack build is LATEST & the latest version is diff from the installed version
273              //3- the pack build is neither LATEST or RECOMMENDED & the pack build is diff from the installed version
274              boolean requestInstall = false;
275              if (pack.getBuild().equalsIgnoreCase(InstalledPack.RECOMMENDED) && pack.getPackInfo().getRecommended() != null && !pack.getPackInfo().getRecommended().equalsIgnoreCase(installedVersion.getVersion()))
276                  requestInstall = true;
277              else if (pack.getBuild().equalsIgnoreCase(InstalledPack.LATEST) && pack.getPackInfo().getLatest() != null && !pack.getPackInfo().getLatest().equalsIgnoreCase(installedVersion.getVersion()))
278                  requestInstall = true;
279              else if (!pack.getBuild().equalsIgnoreCase(InstalledPack.RECOMMENDED) && !pack.getBuild().equalsIgnoreCase(InstalledPack.LATEST) && !pack.getBuild().equalsIgnoreCase(installedVersion.getVersion()))
280                  requestInstall = true;
281  
282              //If the user says yes, update, then force a full install
283              if (requestInstall) {
284      			int result = JOptionPane.showConfirmDialog(this, resources.getString("launcher.install.query"), resources.getString("launcher.install.query.title"), JOptionPane.YES_NO_OPTION, JOptionPane.INFORMATION_MESSAGE);
285  
286      			if (result == JOptionPane.YES_OPTION) {
287      				forceInstall = true;
288      			}
289              }
290          }
291  
292          //If we're forcing an install, then derive the installation build from the pack build
293          //otherwise, just use the installed version
294          String installBuild = null;
295          if (forceInstall && !pack.isLocalOnly()) {
296              installBuild = pack.getBuild();
297  
298              if (installBuild.equalsIgnoreCase(InstalledPack.RECOMMENDED))
299                  installBuild = pack.getPackInfo().getRecommended();
300              else if (installBuild.equalsIgnoreCase(InstalledPack.LATEST))
301                  installBuild = pack.getPackInfo().getLatest();
302          } else if (installedVersion != null)
303              installBuild = installedVersion.getVersion();
304  
305          if (requiresInstall && installBuild != null && !installBuild.isEmpty()) {
306              installer.justInstall(resources, pack, installBuild, forceInstall, this, installProgress);
307          } else {
308              installer.installAndRun(resources, pack, installBuild, forceInstall, this, installProgress);
309          }
310  
311          installProgress.setVisible(true);
312          installProgressPlaceholder.setVisible(false);
313          userChanged(userModel.getCurrentUser());
314          invalidate();
315      }

317      public void launchCompleted() {
318          if (installer.isCurrentlyRunning()) {
319              EventQueue.invokeLater(this::launchCompleted);
320              return;
321          }
322  
323          installProgress.setVisible(false);
324          installProgressPlaceholder.setVisible(true);
325  
326  
327          userModel.setCurrentUser(userModel.getCurrentUser());
328  
329          invalidate();
330  
331      }

333      protected void openModpackOptions(ModpackModel model) {
334  
335          if (modpackOptionsDialog == null) {
336              centralPanel.setTintActive(true);
337              footer.setTintActive(true);
338              modpackOptionsDialog = new ModpackOptionsDialog(this, directories, model, resources);
339              modpackOptionsDialog.setVisible(true);
340              modpackOptionsDialog = null;
341              centralPanel.setTintActive(false);
342              footer.setTintActive(false);
343              modpackPanel.setModpack(model);
344              modpackSelector.forceRefresh();
345          }
346      }
347  
348      protected void refreshModpackOptions(ModpackModel model) {
349          if (modpackOptionsDialog != null)
350              modpackOptionsDialog.refresh(model);
351      }

622      public void setupPlayButtonText(ModpackModel modpack, IUserType user) {
623          playButton.setEnabled(true);
624          playButton.setForeground(LauncherFrame.COLOR_BUTTON_BLUE);
625  
626          if (installer.isCurrentlyRunning()) {
627              playButton.setText(resources.getString("launcher.pack.cancel"));
628          } else if (modpack.getInstalledVersion() != null) {
629              if (userModel.getCurrentUser() == null || userModel.getCurrentUser().isOffline()) {
630                  playButton.setText(resources.getString("launcher.pack.launch.offline"));
631              } else {
632                  playButton.setText(resources.getString("launcher.pack.launch"));
633              }
634              playButton.setIcon(new ImageIcon(resources.colorImage(resources.getImage("play_button.png"), LauncherFrame.COLOR_BUTTON_BLUE)));
635              playButton.setHoverIcon(new ImageIcon(resources.colorImage(resources.getImage("play_button.png"), LauncherFrame.COLOR_BLUE)));
636          } else {
637              if (userModel.getCurrentUser() == null || userModel.getCurrentUser().isOffline()) {
638                  playButton.setEnabled(false);
639                  playButton.setForeground(LauncherFrame.COLOR_GREY_TEXT);
640                  playButton.setText(resources.getString("launcher.pack.cannotinstall"));
641              } else {
642                  playButton.setText(resources.getString("launcher.pack.install"));
643              }
644              playButton.setIcon(new ImageIcon(resources.colorImage(resources.getImage("download_button.png"), LauncherFrame.COLOR_BUTTON_BLUE)));
645              playButton.setHoverIcon(new ImageIcon(resources.colorImage(resources.getImage("download_button.png"), LauncherFrame.COLOR_BLUE)));
646          }
