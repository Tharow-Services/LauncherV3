283      public void save() {
284          String json = Utils.getGson().toJson(this);
285  
286          try {
287              FileUtils.writeStringToFile(settingsFile, json, StandardCharsets.UTF_8);
288          } catch (IOException e) {
289              Utils.getLogger().log(Level.WARNING, "Unable to save installed " + settingsFile);
290          }
291      }
292  
293      protected void buildTechnicRoot() {
294          if (directory == null || directory.isEmpty() || directory.equalsIgnoreCase("portable"))
295              technicRoot = settingsFile.getParentFile();
296          else
297              technicRoot = new File(directory);
298  
299          if (!technicRoot.exists())
300              technicRoot.mkdirs();
301      }
