19   package net.tharow.tantalum.launcher.settings;
20   
21   import com.beust.jcommander.Parameter;
22   import com.beust.jcommander.converters.FileConverter;
23   import com.beust.jcommander.internal.Lists;
24   import lombok.Getter;
25   
26   import java.io.File;
27   import java.util.ArrayList;
28   import java.util.List;
29   @SuppressWarnings("ALL")
30   public final class StartupParameters {
31       private final String[] args;
32       @Parameter
33       private List<String> parameters = Lists.newArrayList();
34       @Parameter(names = {"-console","-c"}, description = "Shows the console window")
35       private boolean console = false;
36       @Parameter(names = {"-logLevel","-ll"}, description = "Sets Log Level")
37       private int logLevel = 8;
38       @Parameter(names = {"-launcheronly","-lo"}, description = "Starts in launcher mode (rather than update/mover)")
39       private boolean launcher = false;
40       @Parameter(names = {"-launcher"}, description = "Legacy launcher mode (indicates we should do a full update)")
41       private boolean oldLauncher = false;
42       @Parameter(names = {"-moveronly"}, description = "Starts in mover mode (copies recently-downloaded update to originally-run package)")
43       private boolean mover = false;
44       @Parameter(names = {"-mover"}, description = "Legacy mover mode- used to detect old-updater clients trying to update")
45       private boolean oldMover = false;
46       @Parameter(names = {"-update"}, description = "Starts in update mode (closes after downloading updated resources)")
47       private boolean update = false;
48       @Parameter(names = {"-movetarget"}, description = "The path of the originally-run package to copy to")
49       private String moveTarget = null;
50       @Parameter(names = {"-solder","-s"}, description = "An override param for the discover URL")
51       private String solder = null;
52       @Parameter(names = {"-platform","-p"}, description = "Add A Platform To the list")
53       private List<String> platform = new ArrayList<>();
54       @Parameter(names = {"-discover","-d"}, description = "Change Discover Url")
55       private String discover = null;
56       @Parameter(names = {"-overrideRoots","-orca"}, description = "Force the override of root ca Certificates")
57       private boolean overrideRoots = false;
58       @Parameter(names = {"-file","-f"}, description = "Install a modpack from file", converter = FileConverter.class)
59       private File inputFile = null;
60       @Parameter(names = {"-uri","-u"}, description = "Install a modpack from api link")
61       private String inputUri = null;
62       @Parameter(names = {"-blockReboot"}, description = "Prevent rebooting the launcher due to bad java properties.")
63       private boolean blockReboot = false;
64       @Parameter(names = {"-buildNumber","-b"}, description = "Force build number to this value for debugging.")
65       private String buildNumber = "";
66       @Parameter(names = {"-offline"}, description = "Force offline mode")
67       private boolean offline = false;
68       @Parameter(names = {"-setting"})
69       @Getter private String setting;
70       @Parameter(names = {"-corca"})
71       private boolean changeOverride = false;
72       @Parameter(names = {"-unlock"}, hidden = true)
73       @Getter private boolean unlocked = false;
74   
75       public StartupParameters(String[] args) {
76           this.args = args;
77       }
78   
79       public List<String> getParameters() {
80           return parameters;
81       }
82   
83       public String[] getArgs() { return args; }
84   
85       public boolean isConsole() {
86           return console;
87       }
88   
89       public boolean isLauncher() { return launcher; }
90   
91       public boolean isLegacyLauncher() { return oldLauncher; }
92   
93       public boolean isMover() { return mover; }
94   
95       public boolean isLegacyMover() { return oldMover; }
96   
97       public boolean isUpdate() { return update; }
98   
99       public boolean isBlockReboot() { return blockReboot; }
100  
101      public boolean isOffline() {return offline;}
102  
103      public String getMoveTarget() { return moveTarget; }
104  
105      public String getSolderUrl() {
106          return solder;
107      }
108  
109      public String getDiscover(){
110          return discover;
111      }
112  
113      public String getBuildNumber() { return buildNumber; }
114  
115      public List<String> getPlatformUrl() {
116          return platform;
117      }
118  
119      public boolean isOverrideRoots() {return overrideRoots;}
120  
121      public File getInputFile() {return inputFile;}
122  
123      public String getInputUri() {return inputUri;}
124  
125      public int getLogLevel() {
126          return logLevel;
127      }
128  
129  
130      public boolean isChangeOverride() {
131          return changeOverride;
132      }
133  }
