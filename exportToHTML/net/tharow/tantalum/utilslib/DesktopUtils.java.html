22   import javax.swing.*;
23   import java.awt.*;
24   import java.io.File;
25   import java.io.IOException;
26   import java.net.URI;
27   import java.net.URISyntaxException;
28   import java.util.logging.Level;
29   
30   public class DesktopUtils {
31       public static void browseUrl(String url) {
32           try {
33               if (url.startsWith("mailto:"))
34                   Desktop.getDesktop().mail(new URI(url));
35               else {
36                   if (Desktop.isDesktopSupported() && Desktop.getDesktop().isSupported(Desktop.Action.BROWSE))
37                       Desktop.getDesktop().browse(new URI(url));
38                   else if(OperatingSystem.getOperatingSystem() == OperatingSystem.LINUX)
39                       Runtime.getRuntime().exec(new String[]{"xdg-open", url});
40                   else if(OperatingSystem.getOperatingSystem() == OperatingSystem.OSX)
41                       Runtime.getRuntime().exec(new String[]{"open", url});
42                   else
43                       JOptionPane.showMessageDialog(null, "Unable to open browser, please visit the URL:\n" + url, "Unable to open browser", JOptionPane.ERROR_MESSAGE);
44               }
45           } catch (IOException | RuntimeException | URISyntaxException ex) {
46               //Thrown by Desktop.browse() - just log & ignore
47               Utils.getLogger().log(Level.SEVERE, ex.getMessage(), ex);
48           } //If we got a bogus URL from the internet, then this will throw.  Log & Ignore
49           //browse() throws a bunch of runtime exceptions if you give it bad input
50           //WHICH IS AWESOME
51   
52       }
53   
54       public static void open(final File file) {
55           new SwingWorker<Void, Void>() {
56               @Override
57               protected Void doInBackground() {
58                   Utils.getLogger().info("Attempting to open "+file.getAbsolutePath());
59                   String asciiUri = file.toURI().toASCIIString();
60                   Utils.getLogger().info("Using "+asciiUri);
61                   if (asciiUri.startsWith("file:") && !asciiUri.startsWith("file://"))
62                       asciiUri = asciiUri.replaceFirst("file:", "file://");
63                   Utils.getLogger().info("Intermediary path "+asciiUri);
64                   try {
65                       Desktop.getDesktop().open(new File(new URI(asciiUri).getPath()));
66                   } catch (Exception ex) {
67                       ex.printStackTrace();
68                   }
69   
70                   return null;
71               }
72           }.execute();
73       }
74   }
