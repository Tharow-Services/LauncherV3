23   import java.lang.management.OperatingSystemMXBean;
24   import java.util.logging.Level;

26   public class Memory {
27       public static final Memory[] memoryOptions = {
28               (new Memory(512, "512 MB", 1)),
29               (new Memory(768, "768 MB", 2)),
30               (new Memory(1024, "1 GB", 0)),
31               (new Memory(1536, "1.5 GB", 3)),
32               (new Memory(2048, "2 GB", 4)),
33               (new Memory(2560, "2.5 GB", 19)),
34               (new Memory(3072, "3 GB", 5)),
35               (new Memory(3584, "3.5 GB", 20)),
36               (new Memory(4096, "4 GB", 6)),
37               (new Memory(5120, "5 GB", 7)),
38               (new Memory(6144, "6 GB", 8)),
39               (new Memory(7168, "7 GB", 9)),
40               (new Memory(8192, "8 GB", 10)),
41               (new Memory(9216, "9 GB", 11)),
42               (new Memory(10240, "10 GB", 12)),
43               (new Memory(11264, "11 GB", 13)),
44               (new Memory(12288, "12 GB", 14)),
45               (new Memory(13312, "13 GB", 15)),
46               (new Memory(14336, "14 GB", 16)),
47               (new Memory(15360, "15 GB", 17)),
48               (new Memory(16384, "16 GB", 18)),
49       };
50       public static final Memory DEFAULT_MEM = memoryOptions[2];
51       public static final int MAX_32_BIT_MEMORY = 1024;
52   
53       public static long getPhysicalMemory() {
54           long maxMemory = 0;
55           try {
56               OperatingSystemMXBean osInfo = ManagementFactory.getOperatingSystemMXBean();
57               if (osInfo instanceof com.sun.management.OperatingSystemMXBean) {
58                   maxMemory = ((com.sun.management.OperatingSystemMXBean) osInfo).getTotalPhysicalMemorySize() / 1024 / 1024;
59               } else {
60                   Utils.getLogger().log(Level.SEVERE, "Failed to detect physical memory: incorrect instance type " + osInfo.getClass().getName());
61               }
62           } catch (Throwable t) {
63               Utils.getLogger().log(Level.SEVERE, "Failed to detect physical memory", t);
64           }
65           return Math.max(512, maxMemory);
66       }
67   
68       public static long getAvailableMemory(boolean is64Bit) {
69           long physical = getPhysicalMemory();
70           if (!is64Bit && physical > MAX_32_BIT_MEMORY)
71               return MAX_32_BIT_MEMORY;
72           return physical;
73       }
74   
75       public static Memory getClosestAvailableMemory(Memory memory, boolean is64Bit) {
76           long available = getAvailableMemory(is64Bit);
77           if (memory.getMemoryMB() <= available)
78               return memory;
79   
80           Memory bestMemory = Memory.memoryOptions[0];
81           for (Memory option : Memory.memoryOptions) {
82               if (option.getMemoryMB() <= available && option.getMemoryMB() > bestMemory.getMemoryMB())
83                   bestMemory = option;
84           }
85   
86           return bestMemory;
87       }
88   
89       final long memory;
90       final String text;
91       final int option;
92   
93       private Memory(int memory, String text, int option) {
94           this.memory = memory;
95           this.text = text;
96           this.option = option;
97       }
98   
99       public long getMemoryMB() {
100          return memory;
101      }
102  
103      public String getDescription() {
104          return text;
105      }
106  
107      public int getSettingsId() {
108          return option;
109      }
110  
111      public static Memory getMemoryFromId(int id) {
112          for (Memory m : memoryOptions) {
113              if (m.getSettingsId() == id) {
114                  return m;
115              }
116          }
117          return DEFAULT_MEM;
118      }
119  
120      public String toString() {
121          return getDescription();
122      }
123  }
