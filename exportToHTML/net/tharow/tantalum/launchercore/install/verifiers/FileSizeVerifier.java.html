20   package net.tharow.tantalum.launchercore.install.verifiers;
21   
22   import net.tharow.tantalum.utilslib.Utils;
23   import org.apache.commons.io.FileUtils;
24   
25   import java.io.File;
26   
27   public class FileSizeVerifier implements IFileVerifier {
28       private final long size;
29   
30       public FileSizeVerifier(long size) {
31           this.size = size;
32       }
33   
34       @Override
35       public boolean isFileValid(File file) {
36           Utils.getLogger().config("File Size Verifier Was Called For "+file.getAbsolutePath());
37           return FileUtils.sizeOf(file) == size;
38       }
39   }
