23   import java.io.IOException;
24   import java.util.zip.ZipException;
25   import java.util.zip.ZipFile;
26   
27   public class ValidZipFileVerifier implements IFileVerifier {
28       @Override
29       public boolean isFileValid(File file) {
30           try (ZipFile zipfile = new ZipFile(file)) {
31               return true;
32           } catch (IOException e) {
33               return false;
34           }
35       }
36   }
