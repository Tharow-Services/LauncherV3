20   package net.tharow.tantalum.launchercore.exception;
21   
22   import java.io.IOException;
23   
24   public class BuildInaccessibleException extends IOException {
25       private final String packDisplayName;
26       private final String build;
27       private Throwable cause;
28       private static final long serialVersionUID = -4905270588640056830L;
29   
30       public BuildInaccessibleException(String displayName, String build) {
31           this.packDisplayName = displayName;
32           this.build = build;
33       }
34   
35       public BuildInaccessibleException(String displayName, String build, Throwable cause) {
36           this(displayName, build);
37           this.cause = cause;
38       }
39   
40       @Override
41       public String getMessage() {
42           if (this.cause != null) {
43               Throwable rootCause = this.cause;
44   
45               while (rootCause.getCause() != null) {
46                   rootCause = rootCause.getCause();
47               }
48   
49               return "An error was raised while attempting to read pack info for modpack " + packDisplayName + ", build " + build + ": " + rootCause.getMessage();
50           } else {
51               return "The pack host returned unrecognizable garbage while attempting to read pack info for modpack " + packDisplayName + ", build " + build + ".";
52           }
53       }
54   
55       @Override
56       public synchronized Throwable getCause() {
57           return cause;
58       }
59   }
