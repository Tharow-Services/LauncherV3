27       public static boolean isJavaVersionAtLeast(String testString, String compareString) {
28           String[] compareVersion = compareString.split("[._]");
29           String[] testVersion = testString.split("[._]");
30   
31           int compareLength = Math.min(compareVersion.length, testVersion.length);
32           for (int i = 0; i < compareLength; i++) {
33               int refVal = Integer.parseInt(compareVersion[i]);
34               int testVal = Integer.parseInt(testVersion[i]);
35   
36               if (refVal == testVal)
37                   continue;
38               return testVal > refVal;
39           }
40   
41           if (compareVersion.length == testVersion.length)
42               return true;
43           return compareVersion.length < testVersion.length;
44       }

62       public IJavaVersion getValidJavaVersion(JavaVersionRepository repository) {
63   
64           boolean requires64Bit = getMemory() > Memory.MAX_32_BIT_MEMORY;
65   
66           IJavaVersion best64Bit = repository.getBest64BitVersion();
67           if (best64Bit != null && isJavaValid(best64Bit.getVersionNumber()))
68               return best64Bit;
69           else if (best64Bit == null && requires64Bit)
70               return null;
71   
72           IJavaVersion bestVersion = null;
73           for (IJavaVersion version : repository.getVersions()) {
74               if (isJavaValid(version.getVersionNumber()) && (!requires64Bit || version.is64Bit()))
75                   bestVersion = version;
76           }
77   
78           return bestVersion;
79       }

81       public Memory getValidMemory(JavaVersionRepository repository) {

84           Memory available = Memory.getClosestAvailableMemory(required, can64Bit);

86           if (available.getMemoryMB() < required.getMemoryMB())

90       }
