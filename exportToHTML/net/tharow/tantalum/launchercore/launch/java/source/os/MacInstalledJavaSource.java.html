20   package net.tharow.tantalum.launchercore.launch.java.source.os;
21   
22   import net.tharow.tantalum.launchercore.launch.java.IVersionSource;
23   import net.tharow.tantalum.launchercore.launch.java.JavaVersionRepository;
24   import net.tharow.tantalum.launchercore.launch.java.version.FileBasedJavaVersion;
25   import net.tharow.tantalum.utilslib.Utils;
26   
27   import java.io.File;
28   
29   public class MacInstalledJavaSource implements IVersionSource {
30       @Override
31       public void enumerateVersions(JavaVersionRepository repository) {
32           repository.addVersion(new FileBasedJavaVersion(new File(getMacJava("1.6"))));
33           repository.addVersion(new FileBasedJavaVersion(new File(getMacJava("1.7"))));
34           repository.addVersion(new FileBasedJavaVersion(new File(getMacJava("1.8"))));
35           repository.addVersion(new FileBasedJavaVersion(new File ("/Library/Internet Plug-Ins/JavaAppletPlugin.plugin/Contents/Home/bin/java")));
36           enumerateJavaHome(repository);
37       }
38   
39       protected String getMacJava(String versionNumber) {
40           String path = Utils.getProcessOutput("/usr/libexec/java_home", "-v", versionNumber);
41           return path + File.separator + "bin" + File.separator + "java";
42       }
43   
44       private void enumerateJavaHome(JavaVersionRepository repository) {
45           String output = Utils.getProcessOutput("/usr/libexec/java_home", "-V");
46   
47           if (output == null || output.isEmpty())
48               return;
49   
50           for (String line : output.split("\\r?\\n")) {
51               // Output is:
52               // Matching Java Virtual Machines (N):
53               //     <version> (<arch>) "<vendor>" - "<runtime name>" <path>
54               // <path> always starts with /Library/
55   
56               int pathIndex = line.indexOf("/Library/");
57   
58               if (pathIndex < 0)
59                   continue;
60   
61               String path = line.substring(pathIndex).trim();
62               repository.addVersion(new FileBasedJavaVersion(new File(path + File.separator + "bin" + File.separator + "java")));
63           }
64       }
65   }
