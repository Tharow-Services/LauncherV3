105      public static Level convert(int value) throws NoSuchElementException
106      {
107          if(!integerLevelMap.containsKey(value)){throw new NoSuchElementException("A Level With Value: "+value+" Couldn't Be Found");}
108          return integerLevelMap.get(value);
109      }
110      public static Level convert(String name) throws NoSuchElementException
111      {
112          if (!stringLevelMap.containsKey(name.toUpperCase().trim())){
113              throw new NoSuchElementException("A Level With Name: "+name.toUpperCase().trim()+" Couldn't Be Found");
114          }
115          return stringLevelMap.getOrDefault(name.toUpperCase().trim(), Level.INFO_ITALICS);
116      }

118      public static Level convert(MC_LEVELS level){
119          switch (level) {
120              case FATAL: return FATAL;
121              case INFO: return INFO;
122              case ERROR: return SEVERE;
123              case WARN: return WARNING;
124              case TRACE: return FINER;
125              case DEBUG: return DEBUG;
126              default: throw new RuntimeException("Tried To Convert Unknown Level In Protected Method");
127          }
128      }

130      public static Level convertMC(String name){
131          try {
132              MC_LEVELS.valueOf(name);
133          } catch (Exception ignored){
134              Utils.getLogger().warning("A Minecraft Level With Name: "+name+" Couldn't Be Found");
135              return Level.SEVERE;
136          }
137          return convert(MC_LEVELS.valueOf(name));
138      }
