24   import net.tharow.tantalum.launchercore.install.ITasksQueue;
25   import net.tharow.tantalum.launchercore.install.InstallTasksQueue;
26   import net.tharow.tantalum.launchercore.install.LauncherDirectories;
27   import net.tharow.tantalum.launchercore.install.tasks.IInstallTask;
28   import net.tharow.tantalum.launchercore.modpacks.ModpackModel;
29   import net.tharow.tantalum.minecraftcore.MojangUtils;
30   import net.tharow.tantalum.minecraftcore.mojang.version.MojangVersion;
31   import net.tharow.tantalum.minecraftcore.mojang.version.MojangVersionBuilder;
32   import net.tharow.tantalum.minecraftcore.mojang.version.builder.FileVersionBuilder;
33   import net.tharow.tantalum.minecraftcore.mojang.version.builder.retrievers.ZipFileRetriever;
34   import net.tharow.tantalum.minecraftcore.mojang.version.io.Artifact;
35   import net.tharow.tantalum.minecraftcore.mojang.version.io.Downloads;
36   import net.tharow.tantalum.minecraftcore.mojang.version.io.Library;

38   import java.io.File;
39   import java.io.IOException;
40   
41   public class HandleVersionFileTask implements IInstallTask {
42       private final ModpackModel pack;
43       private final LauncherDirectories directories;
44       private final ITasksQueue checkLibraryQueue;
45       private final ITasksQueue downloadLibraryQueue;
46       private final ITasksQueue copyLibraryQueue;
47       private final ITasksQueue checkNonMavenLibsQueue;
48       private final MojangVersionBuilder versionBuilder;
49   
50       private String libraryName;
51   
52       public HandleVersionFileTask(ModpackModel pack, LauncherDirectories directories, ITasksQueue checkNonMavenLibsQueue, ITasksQueue checkLibraryQueue, ITasksQueue downloadLibraryQueue, ITasksQueue copyLibraryQueue, MojangVersionBuilder versionBuilder) {
53           this.pack = pack;
54           this.directories = directories;
55           this.checkLibraryQueue = checkLibraryQueue;
56           this.downloadLibraryQueue = downloadLibraryQueue;
57           this.copyLibraryQueue = copyLibraryQueue;
58           this.checkNonMavenLibsQueue = checkNonMavenLibsQueue;
59           this.versionBuilder = versionBuilder;
60       }
61   
62       @Override
63       public String getTaskDescription() {
64           if (libraryName == null)
65               return "Processing version.";
66           else
67               return "Verifying " + libraryName + ".";
68       }
69   
70       @Override
71       public float getTaskProgress() {
72           return 0;
73       }
74   
75       @Override
76       public void runTask(InstallTasksQueue queue) throws IOException, InterruptedException {
77           MojangVersion version = versionBuilder.buildVersionFromKey(null);
78   
79           if (version == null) {
80               throw new DownloadException("The version.json file was invalid.");
81           }
82   
83           // if MC < 1.6, we inject LegacyWrapper
84           // HACK
85           boolean isLegacy = MojangUtils.isLegacyVersion(version.getId());
86   
87           if (isLegacy) {
88               Library legacyWrapper = new Library();
89               legacyWrapper.setName("net.tharow.tantalum:legacywrapper:1.2.1");
90               legacyWrapper.setUrl(TantalumConstants.FORGE_REPO_URL);
91   
92               version.addLibrary(legacyWrapper);
93   
94               version.setMainClass("net.tharow.tantalum.legacywrapper.Launch");
95           }
96   
97           // In Forge 1.13+ and 1.12.2 > 2847, there's an installer jar, and the universal jar can't be used since it
98           // doesn't have the required dependencies to actually launch the game.
99           // So, for Forge 1.13+ we need to use ForgeWrapper to install Forge (it performs deobfuscation at install time).
100          // For Forge 1.12.2 it launches directly, as long as we inject the universal jar (it won't launch at all if
101          // you try running it through ForgeWrapper).
102          final boolean hasModernForge = MojangUtils.hasModernForge(version);
103  
104          if (hasModernForge) {
105              File profileJson = new File(pack.getBinDir(), "install_profile.json");
106              ZipFileRetriever zipVersionRetriever = new ZipFileRetriever(new File(pack.getBinDir(), "modpack.jar"));
107              MojangVersion profileVersion = new FileVersionBuilder(profileJson, zipVersionRetriever, null).buildVersionFromKey("install_profile");
108  
109              final String[] versionIdParts = version.getId().split("-", 3);
110              final boolean is1_12_2 = versionIdParts[0].equals("1.12.2");
111  
112              for (Library library : profileVersion.getLibrariesForOS()) {
113                  if (library.isForge()) {
114                      // For Forge 1.12.2 > 2847, we have to inject the universal jar as a dependency
115                      if (is1_12_2) {
116                          library.setName(library.getName() + ":universal");
117                          library.setUrl(TantalumConstants.forgeMavenRepo);
118  
119                          // Add the mutated library
120                          version.addLibrary(library);
121  
122                          checkLibraryQueue.addTask(new InstallVersionLibTask(library, checkNonMavenLibsQueue, downloadLibraryQueue, copyLibraryQueue, pack, directories));
123                      }
124  
125                      // We normally skip Forge because Forge specifies itself as a dependency, for some reason
126                      continue;
127                  }
128  
129                  checkLibraryQueue.addTask(new InstallVersionLibTask(library, checkNonMavenLibsQueue, downloadLibraryQueue, copyLibraryQueue, pack, directories));
130              }
131  
132              // For Forge 1.13+, we inject our ForgeWrapper as a dependency and launch it through that
133              if (!is1_12_2) {
134                  Library forgeWrapper = new Library();
135                  // TODO: add hash validation
136                  forgeWrapper.setName("io.github.zekerzhayard:ForgeWrapper:1.5.1");
137  
138                  version.addLibrary(forgeWrapper);
139  
140                  version.setMainClass("io.github.zekerzhayard.forgewrapper.installer.Main");
141  
142                  for (Library library : version.getLibrariesForOS()) {
143                      if (library.isForge()) {
144                          Library forgeLauncher = new Library();
145                          forgeLauncher.setName(library.getName() + ":launcher");
146                          forgeLauncher.setUrl(TantalumConstants.forgeMavenRepo);
147  
148                          version.addLibrary(forgeLauncher);
149                          checkLibraryQueue.addTask(new InstallVersionLibTask(forgeLauncher, checkNonMavenLibsQueue, downloadLibraryQueue, copyLibraryQueue, pack, directories));
150  
151                          Library forgeUniversal = new Library();
152                          forgeUniversal.setName(library.getName() + ":universal");
153                          forgeUniversal.setUrl(TantalumConstants.forgeMavenRepo);
154  
155                          checkLibraryQueue.addTask(new InstallVersionLibTask(forgeUniversal, checkNonMavenLibsQueue, downloadLibraryQueue, copyLibraryQueue, pack, directories));
156  
157                          break;
158                      }
159                  }
160              }
161          }
162  
163          for (Library library : version.getLibrariesForOS()) {
164              // If minecraftforge is described in the libraries, skip it
165              // HACK - Please let us get rid of this when we move to actually hosting forge,
166              // or at least only do it if the users are sticking with modpack.jar
167              if (library.isForge()) {
168                  continue;
169              }
170  
171              if (isLegacy && library.getName().startsWith("net.minecraft:launchwrapper")) {
172                  continue;
173              }
174  
175              // Log4j vulnerability patch - CVE-2021-44228
176              // A hotfixed version of 2.0-beta9 for MC 1.7 - 1.12
177              // And version 2.16.0 for >= 1.12
178              if (library.isLog4j()) {
179                  final String[] libNameParts = library.getName().split(":");
180                  // org.apache.logging.log4j:log4j-core:2.0-beta9
181                  // org.apache.logging.log4j    log4j-core        2.0-beta9
182                  // Determine what version we need
183                  String log4jVersion = "2.16.0";
184                  if (libNameParts[2].equals("2.0-beta9")) {
185                      log4jVersion = "2.0-beta9-fixed";
186                  }
187                  String sha1;
188                  int size;
189                  String artifactName = libNameParts[1];
190                  switch (log4jVersion) {
191                      case "2.16.0":
192                          switch (artifactName) {
193                              case "log4j-api":
194                                  sha1 = "f821a18687126c2e2f227038f540e7953ad2cc8c";
195                                  size = 301892;
196                                  break;
197                              case "log4j-core":
198                                  sha1 = "539a445388aee52108700f26d9644989e7916e7c";
199                                  size = 1789565;
200                                  break;
201                              case "log4j-slf4j18-impl":
202                                  sha1 = "0c880a059056df5725f5d8d1035276d9749eba6d";
203                                  size = 21249;
204                                  break;
205                              default:
206                                  throw new RuntimeException("Unknown log4j artifact " + artifactName + ", cannot continue");
207                          }
208                          break;
209                      case "2.0-beta9-fixed":
210                          switch (artifactName) {
211                              case "log4j-api":
212                                  sha1 = "b61eaf2e64d8b0277e188262a8b771bbfa1502b3";
213                                  size = 107347;
214                                  break;
215                              case "log4j-core":
216                                  sha1 = "677991ea2d7426f76309a73739cecf609679492c";
217                                  size = 677588;
218                                  break;
219                              default:
220                                  throw new RuntimeException("Unknown log4j artifact " + artifactName + ", cannot continue");
221                          }
222                          break;
223                      default:
224                          throw new RuntimeException("Unknown log4j version " + log4jVersion + ", cannot continue");
225                  }
226                  String url = String.format("https://mirror.technicpack.net/Technic/lib/org/apache/logging/log4j/%1$s/%2$s/%1$s-%2$s.jar", artifactName, log4jVersion);
227                  Library fixedLog4j = new Library();
228                  fixedLog4j.setName("org.apache.logging.log4j:" + artifactName + ":" + log4jVersion);
229                  Artifact artifact = new Artifact(url, sha1, size);
230                  Downloads downloads = new Downloads();
231                  downloads.setArtifact(artifact);
232                  fixedLog4j.setDownloads(downloads);
233  
234                  // Add fixed lib
235                  version.addLibrary(fixedLog4j);
236                  checkLibraryQueue.addTask(new InstallVersionLibTask(fixedLog4j, checkNonMavenLibsQueue, downloadLibraryQueue, copyLibraryQueue, pack, directories));
237  
238                  // Remove unpatched lib
239                  version.removeLibrary(library.getName());
240  
241                  continue;
242              }
243  
244              checkLibraryQueue.addTask(new InstallVersionLibTask(library, checkNonMavenLibsQueue, downloadLibraryQueue, copyLibraryQueue, pack, directories));
245          }
246  
247          //noinspection unchecked
248          queue.setMetadata(version);
249      }
250  }
