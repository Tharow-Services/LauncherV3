78       public MicrosoftUser loginNewUser() throws MicrosoftAuthException {
79           final String tempCredentialId = UUID.randomUUID().toString();
80   
81           Credential credential = getOAuthCredential(tempCredentialId);
82   
83           XboxResponse xboxResponse = authenticateXbox(credential);
84           XboxResponse xstsResponse = authenticateXSTS(xboxResponse.token);
85           XboxMinecraftResponse xboxMinecraftResponse = authenticateMinecraftXbox(xstsResponse);
86           MinecraftProfile profile = getMinecraftProfile(xboxMinecraftResponse);
87   
88           // TODO: what happens if the user changes their Minecraft username?
89           updateCredentialStore(profile.name, credential);
90   
91           return new MicrosoftUser(xboxMinecraftResponse, profile);
92       }

94       public void refreshSession(MicrosoftUser user) throws AuthenticationException {
95           // Refresh the OAuth token (should last 90 inactive days, indefinitely otherwise).
96           // If it fails then we bail.
97           Credential credential = loadExistingCredential(user.getUsername());
98   
99           if (credential == null) {
100              throw new SessionException("Microsoft login expired or is invalid");
101          }
102  
103          try {
104              if (!credential.refreshToken()) {
105                  // Refresh request failed
106                  throw new SessionException("Microsoft login expired or is invalid");
107              }
108          } catch (IOException e) {
109              // A 4xx was received
110              throw new SessionException("Microsoft login expired or is invalid", e);
111          }
112  
113          // Store the updated credential in the filesystem
114          updateCredentialStore(user.getUsername(), credential);
115  
116          // Request the Xbox token
117          XboxResponse xboxResponse = authenticateXbox(credential);
118  
119          // Request the XSTS token
120          XboxResponse xstsResponse = authenticateXSTS(xboxResponse.token);
121  
122          // Request the Mojang token and store it (in memory)
123          XboxMinecraftResponse xboxMinecraftResponse = authenticateMinecraftXbox(xstsResponse);
124          user.updateAuthToken(xboxMinecraftResponse);
125      }

127      private Credential getOAuthCredential(String username) throws MicrosoftAuthException {
128          LocalServerReceiver receiver = new LocalServerReceiver.Builder()
129                  .setHost("localhost").setPort(0).setCallbackPath("/").build();
130  
131          try {
132              AuthorizationCodeFlow flow = buildFlow();
133              return new AuthorizationCodeInstalledApp(flow, receiver, DesktopUtils::browseUrl).authorize(username);
134          } catch (IOException e) {
135              Utils.getLogger().log(Level.SEVERE, "Failed to get OAuth authorization", e);
136              throw new MicrosoftAuthException(OAUTH, "Failed to get OAuth authorization", e);
137          }
138      }

140      private Credential loadExistingCredential(String username) throws MicrosoftAuthException {
141          AuthorizationCodeFlow flow;

143          try {

148              throw new MicrosoftAuthException(OAUTH, "Failed to load from credential store.");
149          }

152      private XboxResponse authenticateXbox(Credential credential) throws MicrosoftAuthException {
153          XboxAuthRequest xboxAuthRequest = new XboxAuthRequest(credential.getAccessToken());
154  
155          HttpContent httpContent = new JsonHttpContent(JSON_FACTORY, xboxAuthRequest);
156          HttpRequest request = buildPostRequest(XBOX_AUTH_URL, httpContent);
157          request.setThrowExceptionOnExecuteError(false);
158          request.setLoggingEnabled(true);
159          request.setCurlLoggingEnabled(true);
160  
161          HttpResponse httpResponse = null;
162          try {
163              httpResponse = request.execute();
164              return httpResponse.parseAs(XboxResponse.class);
165          } catch (IOException e) {
166              Utils.getLogger().log(Level.SEVERE, "Failed to get Xbox authentication.", e);
167              throw new MicrosoftAuthException(XBOX, "Failed to get Xbox authentication.", e);
168          } finally {
169              if (httpResponse != null) {
170                  try {
171                      httpResponse.disconnect();
172                  } catch (IOException ignored) {
173                  }
174              }
175          }
176      }

178      private XboxResponse authenticateXSTS(String token) throws MicrosoftAuthException {
179          XSTSRequest xstsRequest = new XSTSRequest(token);
180  
181          HttpContent httpContent = new JsonHttpContent(JSON_FACTORY, xstsRequest);
182          HttpRequest request = buildPostRequest(XBOX_XSTS_URL, httpContent);
183  
184          // Don't throw on 401, that way we can handle the 401s we expect.
185          request.setThrowExceptionOnExecuteError(false);
186          request.setLoggingEnabled(true);
187          request.setCurlLoggingEnabled(true);
188  
189          HttpResponse httpResponse = null;
190          try {
191              httpResponse = request.execute();
192              httpResponse.setLoggingEnabled(true);
193  
194              switch (httpResponse.getStatusCode()) {
195                  case HttpStatusCodes.STATUS_CODE_UNAUTHORIZED:
196                      XSTSUnauthorized unauthorized = httpResponse.parseAs(XSTSUnauthorized.class);
197                      throw new MicrosoftAuthException(unauthorized.getExceptionType(), "Failed to get XSTS authentication.");
198                  case HttpStatusCodes.STATUS_CODE_OK:
199                      return httpResponse.parseAs(XboxResponse.class);
200                  default:
201                      throw new HttpResponseException(httpResponse);
202              }
203          } catch (IOException e) {
204              Utils.getLogger().log(Level.SEVERE, "Failed to get XSTS authentication.", e);
205              throw new MicrosoftAuthException(XSTS, "Failed to get XSTS authentication.", e);
206          } finally {
207              if (httpResponse != null) {
208                  try {
209                      httpResponse.disconnect();
210                  } catch (IOException ignored) {
211                  }
212              }
213          }
214      }

241      private MinecraftProfile getMinecraftProfile(XboxMinecraftResponse xboxMinecraftResponse) throws MicrosoftAuthException {
242          HttpRequest request = buildGetRequest();
243          String authorization = xboxMinecraftResponse.getAuthorization();
244          request.setHeaders(request.getHeaders().setAuthorization(authorization));
245  
246          // Profile API returns 404 when it returns a response for no purchased account
247          request.setThrowExceptionOnExecuteError(false);
248          request.setLoggingEnabled(true);
249          request.setCurlLoggingEnabled(true);
250  
251          HttpResponse httpResponse = null;
252          try {
253              httpResponse = request.execute();
254              httpResponse.setLoggingEnabled(true);
255  
256              switch (httpResponse.getStatusCode()) {
257                  case HttpStatusCodes.STATUS_CODE_NOT_FOUND:
258                      MinecraftError minecraftError = httpResponse.parseAs(MinecraftError.class);
259                      throw new MicrosoftAuthException(NO_MINECRAFT, "Minecraft Account Error: " + minecraftError.error);
260                  case HttpStatusCodes.STATUS_CODE_OK:
261                      return httpResponse.parseAs(MinecraftProfile.class);
262                  default:
263                      throw new HttpResponseException(httpResponse);
264              }
265          } catch (IOException e) {
266              Utils.getLogger().log(Level.SEVERE, "Failed to load minecraft profile.", e);
267              throw new MicrosoftAuthException(MINECRAFT_PROFILE, "Failed to load minecraft profile.", e);
268          } finally {
269              if (httpResponse != null) {
270                  try {
271                      httpResponse.disconnect();
272                  } catch (IOException ignored) {
273                  }
274              }
275          }
276      }

278      private AuthorizationCodeFlow buildFlow() throws IOException {
279          return new AuthorizationCodeFlow.Builder(
280                          BearerToken.authorizationHeaderAccessMethod(),
281                          HTTP_TRANSPORT,
282                          JSON_FACTORY,
283                          new GenericUrl(TOKEN_SERVER_URL),
284                          new ClientParametersAuthentication(TECHNIC_CLIENT_ID, null),
285                          TECHNIC_CLIENT_ID,
286                          AUTHORIZATION_SERVER_URL)
287                          .setScopes(Arrays.asList(SCOPES))
288                          .setDataStoreFactory(DATA_STORE_FACTORY)
289  //                            .enablePKCE() TODO: Figure out PKCE
290                          .build();
291      }

293      private void updateCredentialStore(String username, Credential credential) {
294          try {
295              // Remove all the other credentials
296              DATA_STORE_FACTORY.getDataStore(StoredCredential.DEFAULT_DATA_STORE_ID).clear();
297  
298              // Store the new one
299              DATA_STORE_FACTORY.getDataStore(StoredCredential.DEFAULT_DATA_STORE_ID)
300                      .set(username, new StoredCredential(credential));
301          } catch (IOException e) {
302              e.printStackTrace();
303              Utils.getLogger().log(Level.WARNING, "Failed to save user credential to the data store.");
304          }
305      }
