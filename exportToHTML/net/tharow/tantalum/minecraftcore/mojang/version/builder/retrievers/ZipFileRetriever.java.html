20   package net.tharow.tantalum.minecraftcore.mojang.version.builder.retrievers;
21   
22   import net.tharow.tantalum.minecraftcore.mojang.version.builder.MojangVersionRetriever;
23   import net.tharow.tantalum.utilslib.ZipUtils;
24   
25   import java.io.File;
26   import java.io.IOException;
27   
28   public class ZipFileRetriever implements MojangVersionRetriever {
29   
30       private final File zip;
31   
32       public ZipFileRetriever(File sourceZip) {
33           this.zip = sourceZip;
34       }
35   
36       @Override
37       public void retrieveVersion(File target, String key) throws IOException, InterruptedException {
38           if (zip.exists()) {
39               ZipUtils.extractFile(zip, target.getParentFile(), target.getName());
40           }
41       }
42   }
