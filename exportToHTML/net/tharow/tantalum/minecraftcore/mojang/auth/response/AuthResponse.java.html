20   package net.tharow.tantalum.minecraftcore.mojang.auth.response;
21   
22   import net.tharow.tantalum.launchercore.auth.IAuthResponse;
23   import net.tharow.tantalum.minecraftcore.mojang.auth.io.Profile;
24   import net.tharow.tantalum.minecraftcore.mojang.auth.io.User;
25   
26   import java.util.Arrays;
27   
28   @SuppressWarnings({"unused"})
29   public class AuthResponse extends Response implements IAuthResponse {
30       private String accessToken;
31       private String clientToken;
32       private Profile[] availableProfiles;
33       private Profile selectedProfile;
34       private User user;
35   
36       public String getAccessToken() {
37           return accessToken;
38       }
39   
40       public String getClientToken() {
41           return clientToken;
42       }
43   
44       public Profile[] getAvailableProfiles() {
45           return availableProfiles;
46       }
47   
48       public net.tharow.tantalum.minecraftcore.mojang.auth.io.Profile getSelectedProfile() {
49           return selectedProfile;
50       }
51   
52       public User getUser() {
53           return user;
54       }
55   
56       @Override
57       public boolean hasError() {
58           return getError() != null;
59       }
60   
61       @Override
62       public String getError() {
63           String error = super.getError();
64   
65           if (this.availableProfiles != null && this.availableProfiles.length == 0 && (error == null || error.isEmpty())) {
66               return "No Minecraft License";
67           } else {
68               return error;
69           }
70       }
71   
72       @Override
73       public String getErrorMessage() {
74           String message = super.getErrorMessage();
75   
76           if (this.availableProfiles != null && this.availableProfiles.length == 0 && (message == null || message.isEmpty())) {
77               return "This Mojang account has no purchased copies of Minecraft attached.";
78           } else {
79               return message;
80           }
81       }
82   
83       @Override
84       public String toString() {
85           return "AuthResponse{" +
86                   "accessToken='" + accessToken + '\'' +
87                   ", clientToken='" + clientToken + '\'' +
88                   ", availableProfiles=" + Arrays.toString(availableProfiles) +
89                   ", selectedProfile=" + selectedProfile +
90                   '}';
91       }
92   }
