120      public static void copyMinecraftJar(File minecraft, File output) throws IOException {
121          String[] security = { "MOJANG_C.DSA",
122                  "MOJANG_C.SF",
123                  "CODESIGN.RSA",
124                  "CODESIGN.SF" };
125          try (JarFile jarFile = new JarFile(minecraft)) {
126              String fileName = jarFile.getName();
127              String fileNameLastPart = fileName.substring(fileName.lastIndexOf(File.separator));
128  
129              JarOutputStream jos = new JarOutputStream(new FileOutputStream(output));
130              Enumeration<JarEntry> entries = jarFile.entries();
131  
132              while (entries.hasMoreElements()) {
133                  JarEntry entry = entries.nextElement();
134                  if (containsAny(entry.getName(), security)) {
135                      continue;
136                  }
137                  InputStream is = jarFile.getInputStream(entry);
138  
139                  //jos.putNextEntry(entry);
140                  //create a new entry to avoid ZipException: invalid entry compressed size
141                  jos.putNextEntry(new JarEntry(entry.getName()));
142                  byte[] buffer = new byte[4096];
143                  int bytesRead = 0;
144                  while ((bytesRead = is.read(buffer)) != -1) {
145                      jos.write(buffer, 0, bytesRead);
146                  }
147                  is.close();
148                  jos.flush();
149                  jos.closeEntry();
150              }
151              jos.close();
152          }
153  
154      }

156      private static boolean containsAny(String inputString, String[] contains) {
157          for (String string : contains) {
158              if (inputString.contains(string)) {
159                  return true;
160              }
161          }
162          return false;
163      }

165      public static MojangVersion parseVersionJson(String json) {
166          JsonObject root = JsonParser.parseString(json).getAsJsonObject();
167  
168          // Safeguard in the event the version.json we get is actually the one inside a vanilla Minecraft jar
169          // (which isn't valid)
170          if (root.has("world_version") && root.has("protocol_version")) {
171              throw new IllegalArgumentException("Invalid version file, this looks like a Minecraft client jar. Are you sure you didn't place a Minecraft jar as the modpack.jar?");
172          }
173  
174          Class<? extends MojangVersion> versionJsonType;
175          if (root.has("minimumLauncherVersion")) {
176              int minLauncherVersion = root.get("minimumLauncherVersion").getAsInt();
177              Map.Entry<Integer, Class<? extends MojangVersion>> entry = versionJsonVersions.floorEntry(minLauncherVersion);
178              if (entry == null) {
179                  throw new IllegalArgumentException("Unsupported minimumLauncherVersion: " + minLauncherVersion);
180              }
181              versionJsonType = entry.getValue();
182          } else { // fallback: check if "arguments" key exists since only 1.13+ should have it
183              versionJsonType = root.has("arguments") ? CompleteVersionV21.class : CompleteVersion.class;
184          }
185          return getGson().fromJson(root, versionJsonType);
186      }

188      public static boolean isLegacyVersion(String version) {
189          final String[] versionParts = version.split("[.-]", 3);
190  
191          return Integer.parseInt(versionParts[0]) == 1 && Integer.parseInt(versionParts[1]) < 6;
192      }

194      public static boolean hasModernForge(MojangVersion version) {
195          boolean foundForge = false;
196          for (Library library : version.getLibrariesForOS()) {
197              if (library.isForge()) {
198                  foundForge = true;
199                  break;
200              }
201          }
202  
203          if (!foundForge) {
204              return false;
205          }
206  
207          Pattern p = Pattern.compile("^(?<mc>[0-9.]+)-forge-(?<forge>[0-9.]+)$");
208          Matcher m = p.matcher(version.getId());
209  
210          if (!m.matches()) {
211              return false;
212          }
213  
214          final String mcVersionString = m.group("mc");
215  
216          final ComparableVersion mcVersion = new ComparableVersion(mcVersionString);
217          final ComparableVersion forgeVersion = new ComparableVersion(m.group("forge"));
218  
219          // The new Forge installer exists in:
220          // Forge for MC 1.13+
221          // Forge for MC 1.12.2, after the version 14.23.5.2847
222  
223          if (mcVersion.compareTo(new ComparableVersion("1.13")) >= 0)
224              return true;
225  
226          return mcVersionString.equals("1.12.2") && forgeVersion.compareTo(new ComparableVersion("14.23.5.2847")) > 0;
227      }

229      public static String getMinecraftVersion(MojangVersion version) {
230          final String id = version.getId();
231  
232          // Simplification in case it doesn't have Forge at all
233          if (!id.contains("-"))
234              return id;
235  
236          // For Forge, this will be "mc-forge"
237          final String[] idParts = id.split("-");
238          return idParts[0];
239      }

241      public static boolean requiresForgeWrapper(MojangVersion version) {
242          if (!hasModernForge(version)) {
243              return false;
244          }
245  
246          final String mcVersion = getMinecraftVersion(version);
247  
248          return !mcVersion.equals("1.12.2");
249      }

251      public static JavaRuntimes getJavaRuntimes() {
252          if (javaRuntimes != null)
253              return javaRuntimes;
254  
255          try {
256              HttpRequest request = REQUEST_FACTORY.buildGetRequest(new GenericUrl(RUNTIMES_URL));
257              HttpResponse httpResponse = request.execute();
258              javaRuntimes = gson.fromJson(httpResponse.parseAsString(), JavaRuntimes.class);
259              return javaRuntimes;
260          } catch (IOException e) {
261              e.printStackTrace();
262              return null;
263          }
264      }
