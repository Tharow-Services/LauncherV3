89       @Override
90       public boolean equals(Object obj) {
91           if (obj == this) return true;
92           if (obj == null || obj.getClass() != this.getClass()) return false;
93           ApplicationLauncher that = (ApplicationLauncher) obj;
94           return Objects.equals(this.platformApi, that.platformApi) &&
95                   Objects.equals(this.directories, that.directories) &&
96                   Objects.equals(this.userModel, that.userModel) &&
97                   Objects.equals(this.javaVersions, that.javaVersions) &&
98                   Objects.equals(this.buildNumber, that.buildNumber);
99       }

129      public GameProcess launch(ModpackModel pack, long memory, LaunchOptions options, ProcessExitListener exitListener, MojangVersion version) throws IOException {
130          List<String> commands = buildCommands(pack, memory, version, options);
131          StringBuilder full = new StringBuilder();
132          boolean first = true;
133  
134          for (String part : commands) {
135              if (!first) full.append(" ");
136              full.append(part);
137              first = false;
138          }
139          Utils.getLogger().info("Running " + full.toString().replace(userModel.getCurrentUser().getAccessToken(), "redacted"));
140          ProcessBuilder processBuilder = new ProcessBuilder(commands).directory(pack.getInstalledDirectory()).redirectErrorStream(true);
141          Map<String, String> envVars = processBuilder.environment();
142          for (String badVar : BAD_ENV_VARS) envVars.remove(badVar);
143          Process process = null;
144          try {
145              process = processBuilder.start();
146          } catch (Exception e) {
147              // We call the exit listener here to ensure that the launcher unhider knows the process exited, otherwise
148              // we get an infinite loop since LauncherFrame.launchCompleted() will reschedule itself if the launcher
149              // unhider considers the process is still running.
150              if (exitListener != null) {
151                  exitListener.onProcessExit();
152              }
153  
154              throw e;
155          }
156          GameProcess mcProcess = new GameProcess(commands, process);
157          if (exitListener != null) mcProcess.setExitListener(exitListener);
158  
159          platformApi.incrementPackRuns(pack.getName());
160          if (!Utils.sendTracking("runModpack", pack.getName(), pack.getInstalledVersion().getVersion(), options.getOptions().getClientId())) {
161              Utils.getLogger().info("Failed to record event");
162          }
163  
164          return mcProcess;
165      }

167      private List<String> buildCommands(ModpackModel pack, long memory, MojangVersion version, LaunchOptions options) {
168          LaunchCommandCollector commands = new LaunchCommandCollector();
169  
170          // Wrapper command (optirun, etc)
171          String wrapperCommand = options.getOptions().getWrapperCommand();
172          if (StringUtils.isNotEmpty(wrapperCommand)) {
173              commands.addRaw(wrapperCommand);
174          }
175  
176          JavaVersion javaVersion = version.getJavaVersion();
177  
178          if (javaVersion != null && options.getOptions().shouldUseMojangJava()) {
179              File runtimeRoot = new File(directories.getRuntimesDirectory(), javaVersion.getComponent());
180              Path javaExecutable;
181              if (OperatingSystem.getOperatingSystem() == OperatingSystem.WINDOWS) {
182                  javaExecutable = runtimeRoot.toPath().resolve("bin/javaw.exe");
183              } else if (OperatingSystem.getOperatingSystem() == OperatingSystem.OSX) {
184                  javaExecutable = runtimeRoot.toPath().resolve("jre.bundle/Contents/Home/bin/java");
185              } else {
186                  javaExecutable = runtimeRoot.toPath().resolve("bin/java");
187              }
188              commands.addRaw(javaExecutable.toString());
189          } else {
190              commands.addRaw(javaVersions.getSelectedPath());
191          }
192  
193          OperatingSystem operatingSystem = OperatingSystem.getOperatingSystem();
194          String nativesDir = new File(pack.getBinDir(), "natives").getAbsolutePath();
195          String cpString = buildClassPath(pack, version);
196  
197          // build arg parameter map
198          Map<String, String> params = new HashMap<>();
199          IUserType user = userModel.getCurrentUser();
200          File gameDirectory = pack.getInstalledDirectory();
201          ILaunchOptions launchOpts = options.getOptions();
202  
203          params.put("auth_username", user.getUsername());
204          params.put("auth_session", user.getSessionId());
205          params.put("auth_access_token", user.getAccessToken());
206  
207          params.put("auth_player_name", user.getDisplayName());
208          params.put("auth_uuid", user.getId());
209  
210  
211          params.put("profile_name", user.getDisplayName());
212          params.put("version_name", version.getId());
213          params.put("version_type", version.getType().getName());
214  
215          params.put("game_directory", gameDirectory.getAbsolutePath());
216          params.put("natives_directory", nativesDir);
217          params.put("classpath", cpString);
218  
219          params.put("resolution_width", Integer.toString(launchOpts.getCustomWidth()));
220          params.put("resolution_height", Integer.toString(launchOpts.getCustomHeight()));
221  
222          StringSubstitutor paramDereferencer = new StringSubstitutor(params);
223  
224          String targetAssets = directories.getAssetsDirectory().getAbsolutePath();
225  
226          String assetsKey = version.getAssetsKey();
227  
228          if (assetsKey == null || assetsKey.isEmpty()) {
229              assetsKey = "legacy";
230          }
231  
232          if (version.getAreAssetsVirtual()) {
233              targetAssets += File.separator + "virtual" + File.separator + assetsKey;
234          } else if (version.getAssetsMapToResources()) {
235              targetAssets = pack.getResourcesDir().getAbsolutePath();
236          }
237  
238          params.put("game_assets", targetAssets);
239          params.put("assets_root", targetAssets);
240          params.put("assets_index_name", assetsKey);
241          params.put("user_type", user.getMCUserType());
242          params.put("user_properties", user.getUserProperties());
243  
244          params.put("launcher_name", "tantalum   ");
245          params.put("launcher_version", "5.1-" + buildNumber.getBuildNumber());
246  
247          // Prepend custom JVM arguments
248          String customJvmArgs = options.getOptions().getJavaArgs();
249          if (StringUtils.isNotEmpty(customJvmArgs)) {
250              customJvmArgs = customJvmArgs.replaceAll("[\\r\\n]", " ");
251              for (String customJvmArg : customJvmArgs.split("[ ]+")) {
252                  commands.addRaw(customJvmArg);
253              }
254          }
255  
256          // build jvm args
257          String launchJavaVersion = javaVersions.getSelectedVersion().getVersionNumber();
258          ArgumentList jvmArgs = version.getJavaArguments();
259  
260          if (jvmArgs != null) {
261              for (String arg : jvmArgs.resolve(options.getOptions(), paramDereferencer)) {
262                  commands.add(arg);
263              }
264          }
265  
266          int permSize = 128;
267          if (memory >= (1024 * 6)) {
268              permSize = 512;
269          } else if (memory >= 2048) {
270              permSize = 256;
271          }
272  
273          if (Objects.equals(user.getUserType(), AuthlibUser.AUTHLIB_USER_TYPE))
274              commands.addRaw("-javaagent:" + directories.getAssetsDirectory().getAbsolutePath() + "\\launcher\\authlib-injector.jar=" + userModel.getAuthlibURL());
275  
276          commands.addRaw("-Xms" + memory + "m");
277          commands.addRaw("-Xmx" + memory + "m");
278  
279          if (!RunData.isJavaVersionAtLeast(launchJavaVersion, "1.8"))
280              commands.add("-XX:MaxPermSize=" + permSize + "m");
281  
282          commands.addUnique("-Djava.library.path=" + nativesDir);
283          // Tell forge 1.5 to download from our mirror instead
284          // This can't be HTTPS because Forge 1.5 doesn't load up Let's Encrypt root certs (we use Let's Encrypt on the mirror)
285          commands.addUnique("-Dfml.core.libraries.mirror=http://mirror.technicpack.net/Technic/lib/fml/%s");
286  
287          // This is required because we strip META-INF from the minecraft.jar
288          commands.addUnique("-Dfml.ignoreInvalidMinecraftCertificates=true");
289          commands.addUnique("-Dfml.ignorePatchDiscrepancies=true");
290  
291          // This is for ForgeWrapper >= 1.4.2
292          if (MojangUtils.requiresForgeWrapper(version)) {
293              commands.addUnique("-Dforgewrapper.librariesDir=" + directories.getCacheDirectory().getAbsolutePath());
294  
295              // The Forge installer jar is really the modpack.jar
296              File modpackJar = new File(pack.getBinDir(), "modpack.jar");
297              commands.addUnique("-Dforgewrapper.installer=" + modpackJar.getAbsolutePath());
298  
299              // We feed ForgeWrapper the unmodified Minecraft jar here
300              String mcVersion = MojangUtils.getMinecraftVersion(version);
301              File minecraftJar = new File(directories.getCacheDirectory(), "minecraft_" + mcVersion + ".jar");
302              commands.addUnique("-Dforgewrapper.minecraft=" + minecraftJar.getAbsolutePath());
303          }
304  
305          commands.addUnique("-Dminecraft.applet.TargetDirectory=" + pack.getInstalledDirectory().getAbsolutePath());
306          commands.addUnique("-Duser.language=en");
307  
308          if (!options.getOptions().shouldUseStencilBuffer())
309              commands.add("-Dforge.forceNoStencil=true");
310  
311          if (operatingSystem.equals(OperatingSystem.OSX)) {
312              commands.add("-Xdock:icon=" + options.getIconPath());
313              commands.add("-Xdock:name=" + pack.getDisplayName());
314          } else if (operatingSystem.equals(OperatingSystem.WINDOWS)) {
315              // I have no idea if this helps technic or not.
316              commands.addUnique("-XX:HeapDumpPath=MojangTricksIntelDriversForPerformance_javaw.exe_minecraft.exe.heapdump");
317          }
318  
319          // build game args
320          commands.addUnique("-cp", cpString);
321          commands.addRaw(version.getMainClass());
322          List<String> mcArgs = version.getMinecraftArguments().resolve(launchOpts, paramDereferencer);
323  
324          // We manually iterate over the arguments so we can add them in "--name value" pairs, and exclude duplicates
325          // this way. For example: --username Foobar
326          ListIterator<String> mcArgsIterator = mcArgs.listIterator();
327          String current, next;
328          while (mcArgsIterator.hasNext()) {
329              current = mcArgsIterator.next();
330  
331              if (mcArgsIterator.hasNext()) {
332                  // We don't consume it so we can later add it to the commands if it turns out to not be a value
333                  // This allows us to keep the pair detection working correctly, and if we do end up using the value,
334                  // we just consume it from the iterator later on
335                  next = mcArgs.get(mcArgsIterator.nextIndex());
336              } else {
337                  next = null;
338              }
339  
340              // For 1.2.1 through 1.5.2, not all of it is a named argument (--argument):
341              // ${auth_player_name} ${auth_session} --gameDir ${game_directory} --assetsDir ${game_assets}
342              if (current.startsWith("--")) {
343                  // This is an --argument, now we check if it has a value or not
344                  if (next != null && !next.startsWith("--")) {
345                      // This --argument has a value, so we add that --argument value pair
346  
347                      // Special case for --tweakClass to allow multiple values of it. This allows Forge and
348                      // LiteLoader to coexist
349                      if (current.equals("--tweakClass")) {
350                          commands.add(current, next);
351                      } else {
352                          commands.addUnique(current, next);
353                      }
354  
355                      // Consume the next element in the iterator
356                      mcArgsIterator.next();
357                  } else {
358                      // This --argument doesn't have a value, so we just add the argument itself as a unique parameter
359                      commands.addUnique(current);
360                  }
361              } else {
362                  // Doesn't start with --, so this is not a named argument, and since we have no way of knowing if this
363                  // is a duplicate or not, we just add it as a regular argument (no duplicate detection)
364                  commands.add(current);
365              }
366          }
367  
368          options.appendToCommands(commands);
369  
370          //TODO: Add all the other less important commands
371          return commands.collect();
372      }

374      private String buildClassPath(ModpackModel pack, MojangVersion version) {
375          StringBuilder result = new StringBuilder();
376          String separator = System.getProperty("path.separator");
377  
378          // if MC < 1.6, we inject LegacyWrapper
379          // HACK
380          boolean isLegacy = MojangUtils.isLegacyVersion(version.getId());
381  
382          final boolean hasModernForge = MojangUtils.hasModernForge(version);
383          final String[] versionIdParts = version.getId().split("-");
384          final boolean is1_12_2 = versionIdParts[0].equals("1.12.2");
385  
386          // Add all the libraries to the classpath.
387          for (Library library : version.getLibrariesForOS()) {
388              if (library.getNatives() != null) {
389                  continue;
390              }
391  
392              // If minecraftforge is described in the libraries, skip it
393              // HACK - Please let us get rid of this when we move to actually hosting forge,
394              // or at least only do it if the users are sticking with modpack.jar
395              if (library.isForge()) {
396                  if (hasModernForge) {
397                      if (!is1_12_2 && !library.getName().endsWith(":launcher")) {
398                          continue;
399                      } else if (is1_12_2 && !library.getName().endsWith(":universal")) {
400                          continue;
401                      }
402                  } else {
403                      continue;
404                  }
405              }
406  
407              if (isLegacy && library.getName().startsWith("net.minecraft:launchwrapper")) {
408                  continue;
409              }
410  
411              File file = new File(directories.getCacheDirectory(), library.getArtifactPath().replace("${arch}", JavaUtils.getJavaBitness()));
412              if (!file.isFile() || !file.exists()) {
413                  throw new RuntimeException("Library " + library.getName() + " not found.");
414              }
415  
416              if (result.length() > 1) {
417                  result.append(separator);
418              }
419              result.append(file.getAbsolutePath());
420          }
421  
422          // Add the modpack.jar to the classpath, if it exists and minecraftforge is not a library already
423          if (!(MojangUtils.hasModernForge(version))) {
424              File modpack = new File(pack.getBinDir(), "modpack.jar");
425              if (modpack.exists()) {
426                  if (result.length() > 1) {
427                      result.append(separator);
428                  }
429                  result.append(modpack.getAbsolutePath());
430              }
431          }
432  
433          // Add the minecraft jar to the classpath
434          File minecraft = new File(pack.getBinDir(), "minecraft.jar");
435          if (!minecraft.exists()) {
436              throw new RuntimeException("Minecraft not installed for this pack: " + pack);
437          }
438          if (result.length() > 1) {
439              result.append(separator);
440          }
441          result.append(minecraft.getAbsolutePath());
442  
443          return result.toString();
444      }
