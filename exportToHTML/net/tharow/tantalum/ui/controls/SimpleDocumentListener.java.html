1    package net.tharow.tantalum.ui.controls;
2    
3    import javax.swing.event.DocumentEvent;
4    import javax.swing.event.DocumentListener;
5    
6    public class SimpleDocumentListener implements DocumentListener {
7    
8        private final Runnable runnable;
9    
10       /**
11        * A Simple Document Listener
12        *
13        * @param runnable what to run during Doc events
14        */
15       public SimpleDocumentListener(Runnable runnable) {this.runnable = runnable;}
16   
17       /**
18        * Gives notification that there was an insert into the document.  The
19        * range given by the DocumentEvent bounds the freshly inserted region.
20        *
21        * @param e the document event
22        */
23       @Override
24       public void insertUpdate(DocumentEvent e) {
25           this.runnable.run();
26       }
27   
28       /**
29        * Gives notification that a portion of the document has been
30        * removed.  The range is given in terms of what the view last
31        * saw (that is, before updating sticky positions).
32        *
33        * @param e the document event
34        */
35       @Override
36       public void removeUpdate(DocumentEvent e) {
37           this.runnable.run();
38       }
39   
40       /**
41        * Gives notification that an attribute or set of attributes changed.
42        *
43        * @param e the document event
44        */
45       @Override
46       public void changedUpdate(DocumentEvent e) {
47           this.runnable.run();
48       }
49   }
