20   package net.tharow.tantalum.ui.controls;
21   
22   import javax.swing.*;
23   import java.awt.*;
24   
25   public class TintablePanel extends JPanel {
26       private Color tintColor;
27       private boolean tintActive;
28       private JLabel overIcon = null;
29   
30       public TintablePanel() {
31           tintColor = new Color(0,0,0,0);
32       }
33   
34       public Color getTintColor() { return tintColor; }
35       public void setTintColor(Color color) {
36           this.tintColor = color;
37       }
38   
39       public void setOverIcon(ImageIcon image) {
40           if (overIcon != null) {
41               remove(overIcon);
42           }
43   
44           overIcon = new JLabel(image);
45           overIcon.setVisible(false);
46           add(overIcon);
47           revalidate();
48       }
49   
50       public boolean isTintActive() { return tintActive; }
51       public void setTintActive(boolean tintActive) {
52           this.tintActive = tintActive;
53   
54           if (overIcon != null) {
55               overIcon.setVisible(tintActive);
56               EventQueue.invokeLater(this::revalidate);
57           }
58   
59           EventQueue.invokeLater(this::repaint);
60       }
61   
62       @Override
63       public void doLayout() {
64           super.doLayout();
65   
66           if (overIcon != null) {
67               int width = overIcon.getIcon().getIconWidth();
68               int height = overIcon.getIcon().getIconHeight();
69               overIcon.setBounds(getWidth() / 2 - width / 2, getHeight() / 2 - height / 2, width, height);
70           }
71       }
72   
73       @Override
74       public void paint(Graphics graphics) {
75           super.paint(graphics);
76   
77           if (tintActive) {
78               graphics.setColor(getTintColor());
79               graphics.fillRect(0,0,getWidth(),getHeight());
80           }
81       }
82   }
