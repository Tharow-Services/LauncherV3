20   package net.tharow.tantalum.ui.controls.installation;
21   
22   import net.tharow.tantalum.launchercore.util.DownloadListener;
23   
24   import javax.swing.*;
25   import java.awt.*;
26   
27   public class ProgressBar extends JLabel implements DownloadListener {
28       float progressPct;
29       private Color backFillColor = null;
30   
31       public ProgressBar() {
32           super("");
33       }
34   
35       public Color getBackFill() { return backFillColor; }
36       public void setBackFill(Color backFill) { backFillColor = backFill; }
37   
38       @Override
39       public void paint(Graphics g) {
40           Graphics2D g2d = (Graphics2D)g;
41   
42           g2d.setRenderingHint(
43                   RenderingHints.KEY_ANTIALIASING,
44                   RenderingHints.VALUE_ANTIALIAS_ON);
45           g2d.setRenderingHint(
46                   RenderingHints.KEY_TEXT_ANTIALIASING,
47                   RenderingHints.VALUE_TEXT_ANTIALIAS_LCD_HRGB);
48           g2d.setRenderingHint(
49                   RenderingHints.KEY_FRACTIONALMETRICS,
50                   RenderingHints.VALUE_FRACTIONALMETRICS_ON);
51   
52           int width = getWidth()-3;
53           int height = getHeight()-3;
54           int x = 1;
55           int y = 1;
56   
57           Insets insets = new Insets(0,0,0,0);
58   
59           if (getBorder() != null)
60               insets = getBorder().getBorderInsets(this);
61   
62           x += insets.left;
63           y += insets.top;
64   
65           width -= (insets.left + insets.right);
66           height -= (insets.top + insets.bottom);
67   
68           Shape clip = g2d.getClip();
69   
70           if (getBackFill() != null) {
71               g2d.setColor(getBackFill());
72               g2d.clipRect(x, y, width, height);
73               g2d.fillRoundRect(x, y, width, height, height, height);
74           }
75   
76           g2d.setColor(getBackground());
77   
78           float clipWidth = progressPct * (float)width;
79           g2d.clipRect(x, y, (int)clipWidth, height);
80           g2d.fillRoundRect(x, y, width, height, height, height);
81           g2d.setClip(clip);
82   
83           g2d.setColor(getForeground());
84           Stroke stroke = g2d.getStroke();
85           g2d.setStroke(new BasicStroke(2));
86           g2d.drawRoundRect(x, y, width, height, height, height);
87           g2d.setStroke(stroke);
88   
89           int iconWidth = 0;
90           if (this.getIcon() != null) {
91               this.getIcon().paintIcon(this, g2d, x+(height/4), y + (height/2 - this.getIcon().getIconHeight()/2));
92               iconWidth = this.getIcon().getIconWidth() + this.getIconTextGap();
93           }
94   
95           g2d.setFont(getFont());
96           int textY = y + ((height - g2d.getFontMetrics().getHeight()) / 2) + g2d.getFontMetrics().getAscent();
97           int pct = (int)(progressPct * 100);
98           String pctText = pct + "%";
99   
100          int textX = (x+width) - (height/4) - g2d.getFontMetrics().stringWidth(pctText);
101          g2d.drawString(pctText, textX, textY);
102  
103          g2d.clipRect(x, y, textX-x-3, height);
104          g2d.drawString(getText(), x+(height/4)+iconWidth, textY);
105          g2d.setClip(clip);
106      }
107  
108      @Override
109      public Dimension getMaximumSize() {
110          return new Dimension(32000,32000);
111      }
112  
113      @Override
114      public Dimension getMinimumSize() {
115          return new Dimension(0,0);
116      }
117  
118      @Override
119      public Dimension getPreferredSize() {
120          return new Dimension(0,0);
121      }
122  
123      public void setProgressThreadSafe(final String progressText, final float progress) {
124          EventQueue.invokeLater(() -> setProgress(progressText, progress/100.0f));
125      }
126  
127      public void setProgress(String progressText, float progress) {
128          setText(progressText);
129          this.progressPct = progress;
130  
131          EventQueue.invokeLater(this::repaint);
132      }
133  
134      @Override
135      public void stateChanged(String fileName, float progress) {
136          setProgressThreadSafe(fileName, progress);
137      }
138  }
