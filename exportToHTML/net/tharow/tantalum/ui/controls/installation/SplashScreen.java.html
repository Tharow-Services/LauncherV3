20   package net.tharow.tantalum.ui.controls.installation;
21   
22   import net.tharow.tantalum.utilslib.Utils;
23   
24   import javax.swing.*;
25   import java.awt.*;
26   import java.awt.image.BufferedImage;
27   
28   public class SplashScreen extends JFrame {
29       protected final ImageIcon image;
30       private ProgressBar progressBar = null;
31   
32       public SplashScreen(Image img, int barHeight) {
33           setUndecorated(true);
34           setTitle("Technic Launcher");
35   
36           this.image = new ImageIcon(img);
37   
38           Container container = getContentPane();
39           container.setLayout(new BorderLayout());
40   
41           // Redraw the image to fix the alpha channel
42           BufferedImage alphaImage = new BufferedImage(image.getIconWidth(), image.getIconHeight(), BufferedImage.TYPE_INT_ARGB);
43           Graphics2D g = alphaImage.createGraphics();
44           g.drawImage(img, 0, 0, image.getIconWidth(), image.getIconHeight(), null);
45           g.dispose();
46   
47           // Draw the image
48           JButton background = new JButton(new ImageIcon(alphaImage));
49           background.setRolloverEnabled(true);
50           background.setRolloverIcon(background.getIcon());
51           background.setSelectedIcon(background.getIcon());
52           background.setDisabledIcon(background.getIcon());
53           background.setPressedIcon(background.getIcon());
54           background.setFocusable(false);
55           background.setContentAreaFilled(false);
56           background.setBorderPainted(false);
57           background.setOpaque(false);
58           container.add(background, BorderLayout.CENTER);
59   
60           if (barHeight > 0) {
61               JPanel panel = new JPanel();
62               panel.setOpaque(false);
63               container.add(panel, BorderLayout.SOUTH);
64   
65               panel.setLayout(new BorderLayout());
66   
67               JPanel progressPanel = new JPanel();
68               progressPanel.setOpaque(false);
69               progressPanel.setLayout(new BoxLayout(progressPanel, BoxLayout.PAGE_AXIS));
70               panel.add(progressPanel, BorderLayout.CENTER);
71               panel.add(Box.createVerticalStrut(barHeight+barHeight-5), BorderLayout.EAST);
72   
73               progressBar = new ProgressBar();
74               progressPanel.add(Box.createVerticalStrut(barHeight-5));
75               progressPanel.add(progressBar);
76   
77           }
78   
79           // Finalize
80           this.getRootPane().setOpaque(false);
81           try {
82               // Try to set a transparent background, but it isn't always supported
83               this.setBackground(new Color(0, 0, 0, 0));
84           } catch (UnsupportedOperationException ex) {
85               this.setBackground(new Color(0, 0, 0));
86           } catch (IllegalArgumentException ex) {
87               Utils.getLogger().warning("Your desktop environment does not support translucent windows. Technic Launcher will not look as rad for you.");
88           }
89       }
90   
91       public ProgressBar getProgressBar() { return progressBar; }
92   }
