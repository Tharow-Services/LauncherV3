33       protected void paintTabBorder(Graphics g, int tabPlacement,
34                                     int tabIndex,

40       protected void paintTabBackground(Graphics g, int tabPlacement,
41                                         int tabIndex,
42                                         int x, int y, int w, int h,
43                                         boolean isSelected ) {
44           g.setColor(tabPane.getBackgroundAt(tabIndex));
45           switch (tabPlacement) {
46               case LEFT : g.fillRect(x + 1, y, w - 1, h - 2); break;
47               case BOTTOM : g.fillRect(x, y, w - 2, h - 1); break;
48               default : g.fillRect(x, y, w - 2, h - 2); break;
49           }
50       }
51   
52       @Override
53       public void paint(Graphics g, JComponent c) {
54           Graphics2D g2d = (Graphics2D)g;
55   
56           g2d.setRenderingHint(
57                   RenderingHints.KEY_ANTIALIASING,
58                   RenderingHints.VALUE_ANTIALIAS_ON);
59           g2d.setRenderingHint(
60                   RenderingHints.KEY_TEXT_ANTIALIASING,
61                   RenderingHints.VALUE_TEXT_ANTIALIAS_LCD_HRGB);
62           g2d.setRenderingHint(
63                   RenderingHints.KEY_FRACTIONALMETRICS,
64                   RenderingHints.VALUE_FRACTIONALMETRICS_ON);
65   
66           super.paint(g, c);
67           g.setColor(tabPane.getBackgroundAt(tabPane.getSelectedIndex()));
68           g.fillRect(tabPane.getX(), 0, tabPane.getWidth(), 2);
69   
70           g.setColor(tabPane.getComponentAt(tabPane.getSelectedIndex()).getBackground());
71           g.fillRect(tabPane.getX(), calculateTabAreaHeight(tabPane.getTabPlacement(), runCount, maxTabHeight) - getTabAreaInsets(0).bottom, tabPane.getWidth(), getTabAreaInsets(0).bottom);
72   
73           drawTriangle(g);
74       }
75   
76       private void drawTriangle(Graphics g) {
77   
78           Polygon triangle = new Polygon();
79           Rectangle tabRect = rects[tabPane.getSelectedIndex()];
80   
81           int y = tabRect.y;
82           int x = tabRect.x;
83           int w = tabRect.width;
84           int h = tabRect.height;
85   
86           switch (tabPane.getTabPlacement()) {
87               case LEFT : {
88                   int height = y + (h - 2) / 2;
89                   triangle.addPoint(x + w, height - 5);
90                   triangle.addPoint(x + w, height + 5);
91                   triangle.addPoint(x + w + 8, height);
92               }
93               case RIGHT : {
94                   int height = y + (h - 2) / 2;
95                   triangle.addPoint(x, height - 5);
96                   triangle.addPoint(x, height + 5);
97                   triangle.addPoint(x - 8, height);
98               }
99               case BOTTOM : {
100                  int width = x + (w - 2) / 2;
101                  triangle.addPoint(width - 5, y);
102                  triangle.addPoint(width + 5, y);
103                  triangle.addPoint(width, y - 8);
104                  break;
105              }
106              default : {
107                  int width = x + (w - 2) / 2;
108                  triangle.addPoint(width - 6, y + h - 2);
109                  triangle.addPoint(width + 6, y + h - 2);
110                  triangle.addPoint(width, y + h + 6);
111              }
112          }
113          g.setColor(tabPane.getBackgroundAt(tabPane.getSelectedIndex()));
114          g.fillPolygon(triangle);
115      }

122      @Override
123      protected Insets getTabAreaInsets(int tabPlacement) {
124          Insets startInsets = new Insets(2, 0, 8, 0);
125          Insets endInsets = new Insets(0,0,0,0);
126          rotateInsets(startInsets, endInsets, tabPlacement);
127          return endInsets;
128      }
