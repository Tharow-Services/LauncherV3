6    public class SimpleMouseListener implements MouseListener {
7        private final Runnable mouseClicked;
8        private final Runnable mousePressed;
9        private final Runnable mouseReleased;
10       private final Runnable mouseEntered;
11       private final Runnable mouseExited;
12

14       /**
15        * Creates The Most Basic on Click Mouse Listener
16        * @param mouseClicked Invoked when the mouse button has been clicked
17        *                     (pressed and released) on a component
18        */
19       public SimpleMouseListener(Runnable mouseClicked){
20           this(mouseClicked, false);
21       }
22   
23       /**
24        * Creates The Most Basic Mouse Listener
25        * @param mouseClicked Invoked when the mouse button has been clicked
26        *                     (pressed and released) on a component
27        * @param forAllEvents Should @param mouseClicked be used for all events
28        *                     default of false
29        */
30       public SimpleMouseListener(Runnable mouseClicked, boolean forAllEvents){
31           if(forAllEvents){
32               this.mouseClicked = mouseClicked;
33               this.mousePressed = mouseClicked;
34               this.mouseReleased = mouseClicked;
35               this.mouseEntered = mouseClicked;
36               this.mouseExited = mouseClicked;
37           } else{
38               this.mouseClicked = mouseClicked;
39               this.mousePressed = null;
40               this.mouseReleased = null;
41               this.mouseEntered = null;
42               this.mouseExited = null;
43           }
44       }
45   
46       /**
47        * Creates a new mouse listener with the selected runnable
48        * Setting Any of the values to null will result in it not being run.
49        * @param mouseClicked Invoked when the mouse button has been clicked (pressed and released) on a component.
50        * @param mousePressed Invoked when a mouse button has been pressed on a component.
51        * @param mouseReleased Invoked when a mouse button has been released on a component.
52        * @param mouseEntered Invoked when the mouse enters a component.
53        * @param mouseExited Invoked when the mouse exits a component.
54        */
55       public SimpleMouseListener(Runnable mouseClicked, Runnable mousePressed, Runnable mouseReleased, Runnable mouseEntered, Runnable mouseExited){
56           this.mouseClicked = mouseClicked;
57           this.mousePressed = mousePressed;
58           this.mouseReleased = mouseReleased;
59           this.mouseEntered = mouseEntered;
60           this.mouseExited = mouseExited;
61       }
62   
63       private void toRun(Runnable runnable){
64           if (runnable != null) {
65               runnable.run();
66           }
67       }
