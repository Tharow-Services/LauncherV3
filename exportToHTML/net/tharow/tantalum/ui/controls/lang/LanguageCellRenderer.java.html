20   package net.tharow.tantalum.ui.controls.lang;
21   
22   import net.tharow.tantalum.launcher.ui.LauncherFrame;
23   import net.tharow.tantalum.ui.lang.ResourceLoader;
24   import net.tharow.tantalum.ui.listitems.LanguageItem;
25   
26   import javax.swing.*;
27   import java.awt.*;
28   
29   public class LanguageCellRenderer  extends JLabel implements ListCellRenderer {
30   
31       private ImageIcon globe;
32   
33       private final Color background;
34       private final Color foreground;
35   
36       public LanguageCellRenderer(ResourceLoader resourceLoader, String langIcon, Color background, Color foreground) {
37   
38           if (langIcon != null)
39               globe = resourceLoader.getIcon(langIcon);
40           this.background = background;
41           this.foreground = foreground;
42   
43           setForeground(foreground);
44           setBackground(background);
45           setFont(resourceLoader.getFont(ResourceLoader.FONT_OPENSANS, 14));
46           setOpaque(true);
47       }
48   
49       @Override
50       public Component getListCellRendererComponent(JList list, Object value, int index, boolean isSelected, boolean cellHasFocus) {
51           setForeground(this.foreground);
52           setBackground(this.background);
53           setFont(((LanguageItem)value).getLanguageResources().getFont(ResourceLoader.FONT_OPENSANS, list.getFont().getSize()));
54           setText(value.toString());
55   
56           Object selectedValue = list.getSelectedValue();
57   
58           if (globe != null) {
59               if (!isSelected && selectedValue != null && selectedValue.equals(value)) {
60                   setIcon(globe);
61               } else {
62                   setIcon(null);
63               }
64           }
65   
66           // Set a lighter background for the currently selected option (on hover)
67           if (selectedValue != null && selectedValue.equals(value)) {
68               setBackground(LauncherFrame.COLOR_SELECTOR_OPTION);
69           } else {
70               setBackground(background);
71           }
72   
73           return this;
74       }
75   }
