20   package net.tharow.tantalum.ui.controls.feeds;
21   
22   import net.tharow.tantalum.ui.lang.ResourceLoader;
23   
24   import javax.swing.*;
25   import java.awt.*;
26   import java.text.NumberFormat;
27   
28   public class StatBox extends JButton {
29       private final ResourceLoader resources;
30   
31       JLabel fieldLabel;
32       JLabel valueLabel;
33   
34       public StatBox(ResourceLoader resources, String fieldName, Integer value) {
35           this.resources = resources;
36           setOpaque(false);
37   
38           initComponents(fieldName);
39   
40           setValue(value);
41       }
42   
43       public void setValue(Integer value) {
44           String valueStr = buildValueStr(value);
45           valueLabel.setText(valueStr);
46       }
47   
48       @Override
49       public void setForeground(Color color) {
50           super.setForeground(color);
51   
52           if (fieldLabel != null)
53               fieldLabel.setForeground(color);
54   
55           if (valueLabel != null)
56               valueLabel.setForeground(color);
57       }
58   
59       @Override
60       protected void paintComponent(Graphics g) {
61           super.paintComponent(g);
62           Dimension arcs = new Dimension(5,5); //Border corners arcs {width,height}, change this to whatever you want
63           int width = getWidth();
64           int height = getHeight();
65           Graphics2D graphics = (Graphics2D) g;
66           graphics.setRenderingHint(RenderingHints.KEY_ANTIALIASING, RenderingHints.VALUE_ANTIALIAS_ON);
67   
68           //Draws the rounded panel with borders.
69           graphics.setColor(getBackground());
70           graphics.fillRoundRect(0, 0, width-1, height-1, arcs.width, arcs.height);//paint background
71       }
72   
73       private void initComponents(String fieldName) {
74           this.setContentAreaFilled(false);
75           this.setFocusPainted(false);
76           this.setCursor(Cursor.getPredefinedCursor(Cursor.HAND_CURSOR));
77           this.setLayout(new BoxLayout(this, BoxLayout.PAGE_AXIS));
78           this.setBorder(BorderFactory.createEmptyBorder(0, 5, 5, 3));
79   
80           valueLabel = new JLabel("");
81           valueLabel.setFont(resources.getFont(ResourceLoader.FONT_OPENSANS, 22, Font.BOLD));
82           valueLabel.setForeground(getForeground());
83           valueLabel.setHorizontalAlignment(SwingConstants.CENTER);
84           valueLabel.setAlignmentX(CENTER_ALIGNMENT);
85           valueLabel.setBorder(BorderFactory.createEmptyBorder(0,0,0,4));
86           this.add(valueLabel);
87   
88           fieldLabel = new JLabel(fieldName);
89           fieldLabel.setFont(resources.getFont(ResourceLoader.FONT_OPENSANS, 12));
90           fieldLabel.setForeground(getForeground());
91           fieldLabel.setHorizontalAlignment(SwingConstants.CENTER);
92           fieldLabel.setAlignmentX(CENTER_ALIGNMENT);
93           this.add(fieldLabel);
94       }
95   
96       private String buildValueStr(Integer value) {        if (value == null)
97               return "??";
98   
99           if (value >= 1000000000) {
100              return buildShortenedStr((value+50000000)/100000000,"B");
101          } else if (value >= 1000000) {
102              return buildShortenedStr((value+50000)/100000, "M");
103          } else if (value >= 1000) {
104              return buildShortenedStr((value+50)/100, "K");
105          } else
106              return NumberFormat.getInstance().format(value);
107      }
108  
109      private String buildShortenedStr(int value, String suffix) {
110          if (value > 100) {
111              return value / 10 +suffix;
112          } else {
113              return (float) value / 10.0f +suffix;
114          }
115      }
116  }
