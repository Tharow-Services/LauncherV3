20   package net.tharow.tantalum.ui.controls.borders;
21   
22   import javax.swing.border.AbstractBorder;
23   import java.awt.*;
24   import java.awt.geom.Area;
25   import java.awt.geom.RoundRectangle2D;
26   
27   public class RoundBorder extends AbstractBorder {
28           private final Color color;
29           private int thickness = 1;
30           private int radii = 8;
31           private Insets insets = null;
32           private BasicStroke stroke = null;
33           private final int strokePad;
34           final RenderingHints hints;
35   
36           public RoundBorder(
37                   Color color) {
38               this(color, 4, 8);
39           }
40   
41           public RoundBorder(
42                   Color color, int thickness, int radii) {
43               this.thickness = thickness;
44               this.radii = radii;
45               this.color = color;
46   
47               stroke = new BasicStroke(thickness);
48               strokePad = thickness / 2;
49   
50               hints = new RenderingHints(
51                       RenderingHints.KEY_ANTIALIASING,
52                       RenderingHints.VALUE_ANTIALIAS_ON);
53   
54               int pad = strokePad;
55               int bottomPad = Math.max(pad + strokePad, 1);
56   
57               int radiusBounds = Math.max(pad + radii/2, 1);
58               insets = new Insets(pad, radiusBounds, bottomPad, radiusBounds);
59           }
60   
61           @Override
62           public Insets getBorderInsets(Component c) {
63               return insets;
64           }
65   
66           @Override
67           public Insets getBorderInsets(Component c, Insets insets) {
68               return getBorderInsets(c);
69           }
70   
71           @Override
72           public void paintBorder(Component c, Graphics g, int x, int y, int width, int height) {
73   
74               Graphics2D g2 = (Graphics2D) g;
75   
76               int bottomLineY = height - thickness;
77   
78               RoundRectangle2D.Double bubble = new RoundRectangle2D.Double(
79                       strokePad,
80                       strokePad,
81                       width - thickness - 1,
82                       bottomLineY,
83                       radii,
84                       radii);
85   
86               Polygon pointer = new Polygon();
87   
88               Area area = new Area(bubble);
89               area.add(new Area(pointer));
90   
91               g2.setRenderingHints(hints);
92   
93               // Paint the BG color of the parent, everywhere outside the clip
94               // of the text bubble.
95               Component parent  = c.getParent();
96               if (parent!=null) {
97                   Color bg = parent.getBackground();
98                   Rectangle rect = new Rectangle(0,0,width, height);
99                   Area borderRegion = new Area(rect);
100                  borderRegion.subtract(area);
101  
102                  Shape oldClip = g2.getClip();
103                  g2.clip(borderRegion);
104                  g2.setColor(bg);
105                  g2.fillRect(0, 0, width, height);
106                  g2.setClip(null);
107              }
108  
109              g2.setColor(color);
110              g2.setStroke(stroke);
111              g2.draw(area);
112          }
113  }
