20   package net.tharow.tantalum.ui.controls.list;
21   
22   import javax.swing.*;
23   import javax.swing.plaf.basic.BasicScrollBarUI;
24   import java.awt.*;
25   
26   public class SimpleScrollbarUI extends BasicScrollBarUI {
27   
28       private final Color trackColor;
29       private final Color thumbColor;
30   
31       public SimpleScrollbarUI(Color trackColor, Color thumbColor) {
32           this.trackColor = trackColor;
33           this.thumbColor = thumbColor;
34       }
35   
36       @Override
37       protected void paintTrack(Graphics g, JComponent c, Rectangle trackBounds) {
38           g.setColor(trackColor);
39           g.fillRect(trackBounds.x, trackBounds.y, trackBounds.width, trackBounds.height);
40       }
41   
42       @Override
43       protected void paintThumb(Graphics g, JComponent c, Rectangle thumbBounds) {
44           g.setColor(thumbColor);
45   
46           int x = thumbBounds.x+1;
47           int y = thumbBounds.y+1;
48           int w = thumbBounds.width-2;
49           int h = thumbBounds.height-2;
50   
51           g.fillRect(x, y, w, h);
52       }
53   
54       @Override
55       protected void installComponents(){
56           super.installComponents();
57           scrollbar.remove(incrButton);
58           scrollbar.remove(decrButton);
59       }
60   
61       @Override
62       protected void layoutVScrollbar(JScrollBar sb)
63       {
64           Dimension sbSize = sb.getSize();
65           Insets sbInsets = sb.getInsets();
66   
67   	/* 
68        * Width and left edge of the buttons and thumb. 
69        */
70           int itemW = sbSize.width - (sbInsets.left + sbInsets.right);
71           int itemX = sbInsets.left;
72           int endTrackY = sbSize.height - sbInsets.bottom;
73   
74           /* The thumb must fit within the height left over after we 
75            * subtract the preferredSize of the buttons and the insets 
76            * and the gaps 
77            */
78           int sbInsetsH = sbInsets.top + sbInsets.bottom;
79           float trackH = sbSize.height - sbInsetsH;
80   
81           /* Compute the height and origin of the thumb.   The case 
82        * where the thumb is at the bottom edge is handled specially 
83        * to avoid numerical problems in computing thumbY.  Enforce 
84        * the thumbs min/max dimensions.  If the thumb doesn't 
85        * fit in the track (trackH) we'll hide it later. 
86        */
87           float min = sb.getMinimum();
88           float extent = sb.getVisibleAmount();
89           float range = sb.getMaximum() - min;
90           float value = sb.getValue();
91   
92           int thumbH = (range <= 0)
93                   ? getMaximumThumbSize().height : (int)(trackH * (extent / range));
94           thumbH = Math.max(thumbH, getMinimumThumbSize().height);
95           thumbH = Math.min(thumbH, getMaximumThumbSize().height);
96   
97           int thumbY = endTrackY - thumbH;
98           if (value < (sb.getMaximum() - sb.getVisibleAmount())) {
99               float thumbRange = trackH - thumbH;
100              thumbY = (int)(0.5f + (thumbRange * ((value - min) / (range - extent))));
101          }
102  
103  	/* Update the trackRect field. 
104       */
105          int itrackY = 0;
106          int itrackH = endTrackY - itrackY;
107          trackRect.setBounds(itemX, itrackY, itemW, itrackH);
108  
109  	/* If the thumb isn't going to fit, zero it's bounds.  Otherwise 
110       * make sure it fits between the buttons.  Note that setting the 
111       * thumbs bounds will cause a repaint. 
112       */
113          if(thumbH >= (int)trackH)	{
114              setThumbBounds(0, 0, 0, 0);
115          }
116          else {
117              if ((thumbY + thumbH) > endTrackY) {
118                  thumbY = endTrackY - thumbH;
119              }
120              if (thumbY  < 0) {
121                  thumbY = 1;
122              }
123              setThumbBounds(itemX, thumbY, itemW, thumbH);
124          }
125      }
126  }
