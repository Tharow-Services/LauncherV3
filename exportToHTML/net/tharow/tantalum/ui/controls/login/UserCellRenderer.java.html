20   package net.tharow.tantalum.ui.controls.login;
21   
22   import net.tharow.tantalum.ui.controls.list.AdvancedCellRenderer;
23   import net.tharow.tantalum.ui.lang.ResourceLoader;
24   import net.tharow.tantalum.launchercore.auth.IUserType;
25   import net.tharow.tantalum.launchercore.image.IImageJobListener;
26   import net.tharow.tantalum.launchercore.image.ImageJob;
27   import net.tharow.tantalum.launchercore.image.ImageRepository;
28   import net.tharow.tantalum.utilslib.ImageUtils;
29   
30   import javax.swing.*;
31   import java.awt.*;
32   import java.util.HashMap;
33   
34   public class UserCellRenderer extends AdvancedCellRenderer<IUserType> implements ListCellRenderer<IUserType>, IImageJobListener<IUserType> {
35       private final Icon addUserIcon;
36   
37       private final ImageRepository<IUserType> mSkinRepo;
38   
39       private static final int ICON_WIDTH = 32;
40       private static final int ICON_HEIGHT = 32;
41   
42       private final HashMap<String, Icon> headMap = new HashMap<>();
43   
44       public UserCellRenderer(ResourceLoader resources, ImageRepository<IUserType> skinRepo) {
45           this.mSkinRepo = skinRepo;
46           addUserIcon = resources.getIcon("add_user.png");
47       }
48   
49       @Override
50       public Component getListCellRendererComponent(JList<? extends IUserType> list, IUserType user, int index, boolean isSelected, boolean cellHasFocus) {
51           super.getListCellRendererComponent(list, user, index, isSelected, cellHasFocus);
52   
53           if (!headMap.containsKey(user.getUsername())) {
54               @SuppressWarnings("unchecked") ImageJob<IUserType> job = mSkinRepo.startImageJob(user);
55               job.addJobListener(this);
56               headMap.put(user.getUsername(), new ImageIcon(ImageUtils.scaleImage(job.getImage(), ICON_WIDTH, ICON_HEIGHT)));
57           }
58   
59           Icon head = headMap.get(user.getUsername());
60   
61           if (head != null) {
62               this.setIcon(head);
63           }
64   
65           return this;
66       }
67   
68       @Override
69       public void jobComplete(ImageJob<IUserType> job) {
70           IUserType user = job.getJobData();
71           headMap.remove(user.getUsername());
72   
73           headMap.put(user.getUsername(), new ImageIcon(ImageUtils.scaleImage(job.getImage(), ICON_WIDTH, ICON_HEIGHT)));
74   
75           this.invalidate();
76       }
77   }
